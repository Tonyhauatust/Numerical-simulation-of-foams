46 {
47  int N = m_system->get_group(m_group_name)->get_size();
48  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
49  // reset forces and torques
50  m_system->reset_forces();
51  m_system->reset_torques();
52 
53  // No need to compute potential, only compute torques in the current configuration
54  if (m_align)
55  m_align->compute();
56  // iterate over all particles
57  for (int i = 0; i < N; i++)
58  {
59  int pi = particles[i];
60  Particle& p = m_system->get_particle(pi);
61  // Update particle position
62  double kappa = m_rng->drnd()-0.5;
63  if (kappa != 0.0)
64  kappa /= fabs(kappa);
65  else
66  kappa = 0.0;
67  kappa *= m_d0;
68  p.x += kappa*p.nx;
69  p.y += kappa*p.ny;
70  p.z += kappa*p.nz;
71  // Project everything back to the manifold
72  m_constrainer->enforce(p);
73  // Update angular velocity
74  p.omega = m_mu*m_constrainer->project_torque(p);
75  //p.omega = m_dt*m_constraint->project_torque(p);
76  // Change orientation of the director (in the tangent plane) according to eq. (1b)
77  double dtheta = m_dt*p.omega + m_stoch_coeff*m_rng->gauss_rng(1.0);
78  //double dtheta = m_dt*m_constraint->project_torque(p) + m_stoch_coeff*m_rng->gauss_rng(1.0);
79  m_constrainer->rotate_director(p,dtheta);
80  //p.omega = dtheta*m_dt;
81  p.age += m_dt;
82  }
