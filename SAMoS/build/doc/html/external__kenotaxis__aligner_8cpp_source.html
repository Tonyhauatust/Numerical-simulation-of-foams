26 void ExternalKenotaxisAlign::compute()
27 {
28  int N = m_system->size();
29  double J = m_J;
30  Mesh& mesh = m_system->get_mesh();
31 
32 
33  // First we compute centre of mass to be able to determine outward direction
34  double xcm = 0.0, ycm = 0.0, zcm = 0.0;
35  int cnt = 0;
36  for (int i = 0; i < N; i++)
37  {
38  Particle& pi = m_system->get_particle(i);
39  if (pi.in_tissue)
40  {
41  xcm += pi.x;
42  ycm += pi.y;
43  zcm += pi.z;
44  cnt++;
45  }
46  }
47  xcm /= cnt; ycm /= cnt; zcm /= cnt;
48 
49  for (int i = 0; i < N; i++)
50  {
51  Particle& pi = m_system->get_particle(i);
52  if (pi.boundary)
53  {
54  // Compute the direction perpendicular to the boundary.
55  // We cheat by using the centre of mass position to deterimine the outside and inside of the tissue.
56 
57  // vector (X, Y, Z) points from centre of mass to particle i
58  double force_sign = 1.0;
59  double X = pi.x - xcm, Y = pi.y - ycm, Z = pi.z - zcm;
60  double len_R = sqrt(X*X + Y*Y + Z*Z);
61  X /= len_R; Y /= len_R; Z /= len_R;
62  // get relative vectors of neighbouring boundary particles
63  Particle& pj = m_system->get_particle(pi.boundary_neigh[0]);
64  Particle& pk = m_system->get_particle(pi.boundary_neigh[1]);
65  double xji = pj.x - pi.x, yji = pj.y - pi.y, zji = pj.z - pi.z;
66  double len_ji = sqrt(xji*xji + yji*yji + zji*zji);
67  xji /= len_ji; yji /= len_ji; zji /= len_ji;
68  double xki = pk.x - pi.x, yki = pk.y - pi.y, zki = pk.z - pi.z;
69  double len_ki = sqrt(xki*xki + yki*yki + zki*zki);
70  xki /= len_ki; yki /= len_ki; zki /= len_ki;
71  // compute vector perpendicular to r_kj
72  double x = -(yji-yki), y = (xji-xki), z = 0.0; // give up with three dimensions, z=0
73  double len_r = sqrt(x*x + y*y + z*z);
74  x /= len_r; y /= len_r; z /= len_r;
75  // compute dot product with vector connecting center of mass and (x,y,z)
76  if ((x*X + y*Y + z*Z) < 0.0)
77  force_sign = -1.0;
78  double J_factor = force_sign*J;
79  double tau_z = pi.nx*y - pi.ny*x;
80  //if (m_has_params)
81  //J = m_type_params[pi.get_type()-1].J
82  //
83 
84  pi.tau_x += 0.0;
85  pi.tau_y += 0.0;
86  pi.tau_z += J_factor*tau_z;
87  }
88  }
89 }
