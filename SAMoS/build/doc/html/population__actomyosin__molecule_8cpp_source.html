42 void PopulationActomyosinMolecule::divide(int t)
43 {
44  if ((m_freq > 0) && (t % m_freq == 0) && (m_detach_rate > 0.0)) // Attempt D to A transition only at certain time steps
45  {
46  if (!m_system->group_ok(m_group_name))
47  {
48  cout << "Before actomyosin molecule detachment: Group info mismatch for group : " << m_group_name << endl;
49  throw runtime_error("Group mismatch.");
50  }
51  int Nmol = m_system->number_of_molecules();
52  // Probability of detachment for a given particle is rate per particle multiplied with time,
53  // where time is equal to m_freq*integrator_time_step.
54  double detach_prob = m_detach_rate*m_freq*m_system->get_integrator_step();
55  // loop over all molecules in the system
56  for (int mol = 0; mol < Nmol; mol++)
57  {
58  vector<int>& particles = m_system->get_mol_particles(mol); // get all particle ids that belong to this molecule
59  bool detach = false;
60  for (unsigned int i = 0; i < particles.size(); i++) // loop over all particles in the molecule
61  {
62  Particle& pi = m_system->get_particle(particles[i]);
63  if (pi.get_type() == m_type_a) // if at least one of the beads is of type "attached", mark for detachment
64  {
65  detach = true;
66  break;
67  }
68  }
69  // if marked for detachment and random number below detachment probability, make transition attached --> detached on all beads
70  if (detach && m_rng->drnd() < detach_prob)
71  for (unsigned int i = 0; i < particles.size(); i++) // loop over all particles in the molecule
72  {
73  Particle& pi = m_system->get_particle(particles[i]);
74  if (pi.get_type() == m_type_a)
75  pi.set_type(m_type_d);
76  }
77  }
78  }
79 }
