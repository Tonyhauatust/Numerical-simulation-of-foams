46 void PopulationRandom::divide(int t)
47 {
48  if (m_freq > 0 && t % m_freq == 0) // Attempt division only at certain time steps
49  {
50  int new_type; // type of new particle
51  double new_r; // radius of newly formed particle
52  int N = m_system->get_group(m_group_name)->get_size();
53  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
54  bool periodic = m_system->get_periodic();
55  BoxPtr box = m_system->get_box();
56  double prob_div = m_div_rate*m_freq*m_system->get_integrator_step(); // actual probability of dividing now: rate * (attempt_freq * dt)
57  if (prob_div > 1.0)
58  {
59  m_msg->msg(Messenger::ERROR,"Division rate "+lexical_cast<string>(prob_div)+" is too large for current time step and attempt rate.");
60  throw runtime_error("Too high division.");
61  }
62  for (int i = 0; i < N; i++)
63  {
64  int pi = particles[i];
65  Particle& p = m_system->get_particle(pi);
66  if (m_rng->drnd() < p.age*prob_div)
67  {
68  Particle p_new(m_system->size(), p.get_type(), p.get_radius());
69  p_new.x = p.x + m_alpha*p.get_radius()*p.nx;
70  p_new.y = p.y + m_alpha*p.get_radius()*p.ny;
71  p_new.z = p.z + m_alpha*p.get_radius()*p.nz;
72  if (periodic)
73  {
74  if (p_new.x > box->xhi) p_new.x -= box->Lx;
75  else if (p_new.x < box->xlo) p_new.x += box->Lx;
76  if (p_new.y > box->yhi) p_new.y -= box->Ly;
77  else if (p_new.y < box->ylo) p_new.y += box->Ly;
78  if (p_new.z > box->zhi) p_new.z -= box->Lz;
79  else if (p_new.z < box->zlo) p_new.z += box->Lz;
80  }
81  p.x -= (1.0-m_alpha)*p.get_radius()*p.nx;
82  p.y -= (1.0-m_alpha)*p.get_radius()*p.ny;
83  p.z -= (1.0-m_alpha)*p.get_radius()*p.nz;
84  if (periodic)
85  {
86  if (p.x > box->xhi) p.x -= box->Lx;
87  else if (p.x < box->xlo) p.x += box->Lx;
88  if (p.y > box->yhi) p.y -= box->Ly;
89  else if (p.y < box->ylo) p.y += box->Ly;
90  if (p.z > box->zhi) p.z -= box->Lz;
91  else if (p.z < box->zlo) p.z += box->Lz;
92  }
93  p_new.nx = p.nx; p_new.ny = p.ny; p_new.nz = p.nz;
94  p_new.vx = p.vx; p_new.vy = p.vy; p_new.vz = p.vz;
95  p_new.Nx = p.Nx; p_new.Ny = p.Ny; p_new.Nz = p.Nz;
96  p.age = 0.0;
97  p_new.age = 0.0;
98  for(list<string>::iterator it_g = p.groups.begin(); it_g != p.groups.end(); it_g++)
99  p_new.add_group(*it_g);
100  if (m_rng->drnd() < m_type_change_prob_1) // Attempt to change type, radius and group for first child
101  {
102  if (m_new_type == 0)
103  new_type = p.get_type();
104  else
105  new_type = m_new_type;
106  p.set_type(new_type);
107  if (m_new_radius == 0.0)
108  new_r = p.get_radius();
109  else
110  new_r = m_new_radius;
111  p.set_radius(new_r);
112  m_system->change_group(p.get_id(),m_old_group,m_new_group);
113  }
114  m_system->add_particle(p_new);
115  Particle& pr = m_system->get_particle(p_new.get_id());
116  if (m_rng->drnd() < m_type_change_prob_2) // Attempt to change type, radius and group for second child
117  {
118  if (m_new_type == 0)
119  new_type = pr.get_type();
120  else
121  new_type = m_new_type;
122  pr.set_type(new_type);
123  if (m_new_radius == 0.0)
124  new_r = pr.get_radius();
125  else
126  new_r = m_new_radius;
127  pr.set_radius(new_r);
128  m_system->change_group(pr.get_id(),m_old_group,m_new_group);
129  }
130  }
131  }
132  m_system->set_force_nlist_rebuild(true);
133  }

144 void PopulationRandom::remove(int t)
145 {
146  if (m_freq > 0 && t % m_freq == 0) // Attempt removal only at certain time steps
147  {
148  int N = m_system->get_group(m_group_name)->get_size();
149  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
150  vector<int> to_remove;
151  double prob_death = m_death_rate*m_freq*m_system->get_integrator_step(); // actual probability of dividing now: rate * (attempt_freq * dt)
152  if (prob_death>1.0)
153  {
154  cout << "Error: death probability " << prob_death << " is too large for current time step and attempt rate!" << endl;
155  throw runtime_error("Too high death.");
156  }
157  for (int i = 0; i < N; i++)
158  {
159  int pi = particles[i];
160  Particle& p = m_system->get_particle(pi);
161  if (m_rng->drnd() < p.age*prob_death)
162  to_remove.push_back(p.get_id());
163  }
164  int offset = 0;
165  for (vector<int>::iterator it = to_remove.begin(); it != to_remove.end(); it++)
166  {
167  m_system->remove_particle((*it)-offset);
168  offset++;
169  }
170  if (m_system->size() == 0)
171  {
172  m_msg->msg(Messenger::ERROR,"Random population control. No particles left in the system. Please reduce that death rate.");
173  throw runtime_error("No particles left in the system.");
174  }
175  if (!m_system->group_ok(m_group_name))
176  {
177  cout << "Remove P: Group info mismatch for group : " << m_group_name << endl;
178  throw runtime_error("Group mismatch.");
179  }
180  m_system->set_force_nlist_rebuild(true);
181  }
