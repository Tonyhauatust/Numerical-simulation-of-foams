42 void IntegratorBrownian::integrate()
43 {
44  int N = m_system->get_group(m_group_name)->get_size();
45  double T = m_temp->get_val(m_system->get_run_step());
46  double B = sqrt(2.0*m_mu*T);
47  double sqrt_dt = sqrt(m_dt);
48  double fd_x, fd_y, fd_z; // Deterministic part of the force
49  double fr_x = 0.0, fr_y = 0.0, fr_z = 0.0; // Random part of the force
50  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
51 
52  // reset forces and torques
53  m_system->reset_forces();
54  m_system->reset_torques();
55 
56  // If nematic, attempt to flip directors
57  if (m_nematic)
58  for (int i = 0; i < N; i++)
59  {
60  int pi = particles[i];
61  Particle& p = m_system->get_particle(pi);
62  if (m_rng->drnd() < m_tau) // Flip direction n with probability m_tua (dt/tau, where tau is the parameter given in the input file).
63  {
64  p.nx = -p.nx; p.ny = -p.ny; p.nz = -p.nz;
65  if (m_velocity)
66  {
67  p.vx = -p.vx; p.vy = -p.vy; p.vz = -p.vz;
68  }
69  }
70  }
71 
72  // compute forces in the current configuration
73  if (m_potential)
74  m_potential->compute(m_dt);
75  // compute torques in the current configuration
76  if (m_align)
77  m_align->compute();
78  // iterate over all particles
79  for (int i = 0; i < N; i++)
80  {
81  int pi = particles[i];
82  Particle& p = m_system->get_particle(pi);
83  // compute deterministic forces
84  fd_x = m_v0*p.nx + m_mu*p.fx;
85  fd_y = m_v0*p.ny + m_mu*p.fy;
86  fd_z = m_v0*p.nz + m_mu*p.fz;
87  // Update velocity
88  p.vx = fd_x;
89  p.vy = fd_y;
90  p.vz = fd_z;
91  // Update particle position according to the eq. (1a)
92  p.x += m_dt*fd_x;
93  p.y += m_dt*fd_y;
94  p.z += m_dt*fd_z;
95  // Check is non-zero T
96  if (T > 0.0)
97  {
98  fr_x = B*m_rng->gauss_rng(1.0);
99  fr_y = B*m_rng->gauss_rng(1.0);
100  fr_z = B*m_rng->gauss_rng(1.0);
101  p.vx += fr_x;
102  p.vy += fr_y;
103  p.vz += fr_z;
104  p.x += sqrt_dt*fr_x;
105  p.y += sqrt_dt*fr_y;
106  p.z += sqrt_dt*fr_z;
107  }
108  // Project everything back to the manifold
109  m_constrainer->enforce(p);
110  // Update angular velocity
111  p.omega = m_mur*m_constrainer->project_torque(p);
112  //p.omega = m_dt*m_constraint->project_torque(p);
113  // Change orientation of the director (in the tangent plane) according to eq. (1b)
114  double dtheta = m_dt*p.omega + m_stoch_coeff*m_rng->gauss_rng(1.0);
115  //double dtheta = m_dt*m_constraint->project_torque(p) + m_stoch_coeff*m_rng->gauss_rng(1.0);
116  m_constrainer->rotate_director(p,dtheta);
117  if (m_velocity)
118  m_constrainer->rotate_velocity(p,dtheta);
119  //p.omega = dtheta*m_dt;
120  p.age += m_dt;
121  }
122  // Update vertex mesh
123  m_system->update_mesh();
124 }
