76 void Mesh::generate_faces()
77 {
78  m_is_triangulation = true;
79  vector<vert_angle> angles;
80  for (int i = 0; i < m_nedge; i++)
81  {
82  Face face = Face(m_nface);
83  Edge& E = m_edges[i];
84  if (!E.visited)
85  {
86  E.visited = true;
87  int seed = E.from;
88  int vn = E.to;
89  face.add_vertex(seed);
90  face.add_vertex(vn);
91  face.add_edge(E.id);
92  int vp = seed;
93  int prev_edge = E.id;
94  while (vn != seed)
95  {
96  Vertex& Vp = m_vertices[vp];
97  Vertex& Vn = m_vertices[vn];
98  Vector3d ri = Vn.r - Vp.r;
99  angles.clear();
100  for (unsigned int e = 0; e < Vn.edges.size(); e++)
101  {
102  Edge& Ej = m_edges[Vn.edges[e]];
103  if (!Ej.visited && Ej.to != vp)
104  {
105  Vector3d rj = m_vertices[Ej.to].r - Vn.r;
106  angles.push_back(make_pair(e,M_PI - angle(ri,rj,Vn.N)));
107  }
108  }
109  sort(angles.begin(),angles.end(),comp);
110  Edge& Ej = m_edges[Vn.edges[angles[0].first]];
111  Ej.visited = true;
112  if (Ej.to != seed) face.add_vertex(Ej.to);
113  face.add_edge(Ej.id);
114  m_edges[prev_edge].next = Ej.id;
115  prev_edge = Ej.id;
116  vp = vn;
117  vn = Ej.to;
118  if (vn == seed) m_edges[prev_edge].next = E.id; // original edge is the "next" edge for last edge in the loop
119  }
120  }
121  if (face.vertices.size() > 0)
122  {
123  if (face.vertices.size() > 3) face.is_hole = true;
124  m_faces.push_back(face);
125  for (unsigned int f = 0; f < face.vertices.size(); f++)
126  m_vertices[face.vertices[f]].add_face(m_nface);
127  for (unsigned int f = 0; f < face.edges.size(); f++)
128  m_edges[face.edges[f]].face = m_nface;
129  m_nface++;
130  }
131  if (face.vertices.size() > 3 && (!face.is_hole))
132  m_is_triangulation = false;
133  }

257 void Mesh::order_star(int v)
258 {
259  Vertex& V = m_vertices[v];
260  V.neigh.clear();
261  V.faces.clear();
262  vector<int> edges;
263 
264  if (V.edges.size() == 0)
265  V.attached = false;
266  else
267  {
268  if (!V.boundary) // For internal vertices just pick the first edge
269  edges.push_back(V.edges[0]);
270  else // For a boundary vertex, first edge in the star is always the one with its pair being a boundary edge
271  {
272  for (unsigned int e = 0; e < V.edges.size(); e++)
273  {
274  Edge& E = m_edges[V.edges[e]];
275  if (m_edges[E.pair].boundary)
276  {
277  edges.push_back(E.id);
278  break;
279  }
280  }
281  }
282  int i = 0;
283  while(edges.size() < V.edges.size())
284  {
285  Edge& Ei = m_edges[edges[i++]];
286  Vertex& Vi = m_vertices[Ei.to];
287  Vector3d ri = Vi.r - V.r;
288  double min_angle = 2.0*M_PI;
289  int next_edge;
290  for (unsigned int e = 0; e < V.edges.size(); e++)
291  {
292  Edge& Ej = m_edges[V.edges[e]];
293  if (find(edges.begin(), edges.end(), Ej.id) == edges.end()) // Check if the edge has not already been included (note that most likely we don't even need this test, since angles are ordered)
294  {
295  Vertex& Vj = m_vertices[Ej.to];
296  Vector3d rj = Vj.r - V.r;
297  double ang = angle(ri,rj,V.N);
298  ang = (ang >= 0.0) ? ang : 2.0*M_PI+ang;
299  if (ang < min_angle)
300  {
301  min_angle = ang;
302  next_edge = Ej.id;
303  }
304  }
305  }
306  edges.push_back(next_edge);
307  }
308  copy(edges.begin(),edges.end(),V.edges.begin());
309  // Here we handle neighbours and faces
310  for (int e = 0; e < V.n_edges; e++)
311  {
312  Edge& E = m_edges[V.edges[e]];
313  V.neigh.push_back(E.to);
314  V.faces.push_back(E.face);
315  }
316 
317  // Vertex star is not ordered
318  V.ordered = true;
319  // Make sure that the star of the boundary is in proper order
320  if (V.boundary)
321  this->order_boundary_star(V.id);
322  this->order_dual(v);
323  }

333 void Mesh::order_dual(int v)
334 {
335  Vertex& V = m_vertices[v];
336  V.dual.clear();
337  V.dual_neighbour_map.clear();
338  if (V.n_faces == 0)
339  V.attached = false;
340  if (!V.attached) return;
341 
342  if (!V.ordered)
343  {
344  cout << V << endl;
345  throw runtime_error("Vertex star has to be ordered before we can order its dual.");
346  }
347 
348  Vector3d fc(0.0,0.0,0.0);
349  for (unsigned int f = 0; f < V.faces.size(); f++)
350  {
351  Face& face = m_faces[V.faces[f]];
352  if (face.is_hole)
353  fc += V.r;
354  else
355  fc += face.rc;
356  }
357  fc.scale(1.0/static_cast<double>(V.faces.size()));
358 
359  // Now we order duals. This is important for proper computation of areas and force
360  if (!V.boundary)
361  {
362  V.dual.push_back(V.faces[0]);
363  V.dual_neighbour_map.push_back(V.neigh[0]);
364  }
365  else
366  {
367  Vector3d r0 = m_vertices[V.neigh[0]].r - V.r;
368  double min_angle = M_PI;
369  int fface = 0;
370  for (unsigned int f = 0; f < V.faces.size(); f++)
371  {
372  Face& face = m_faces[V.faces[f]];
373  if (!face.is_hole)
374  {
375  Vector3d r1 = face.rc - V.r;
376  double ang = angle(r0,r1,V.N);
377  if (ang > -0.5*M_PI && ang < min_angle)
378  {
379  min_angle = ang;
380  fface = f;
381  }
382  }
383  }
384  V.dual.push_back(V.faces[fface]);
385  V.dual_neighbour_map.push_back(V.neigh[fface]);
386  }
387  int i = 0;
388  while(V.dual.size() < V.faces.size())
389  {
390  Face& Fi = m_faces[V.dual[i]];
391  Vector3d ri;
392  if (V.boundary)
393  ri = Fi.rc - V.r;
394  else
395  ri = Fi.rc - fc;
396  double min_angle = 2.0*M_PI;
397  int next_dual;
398  bool can_add = false;
399  int face_index;
400  for (unsigned int f = 0; f < V.faces.size(); f++)
401  {
402  Face& Fj = m_faces[V.faces[f]];
403  if (!Fj.is_hole)
404  {
405  if (find(V.dual.begin(), V.dual.end(), Fj.id) == V.dual.end())
406  {
407  Vector3d rj;
408  if (V.boundary)
409  rj = Fj.rc - V.r;
410  else
411  rj = Fj.rc - fc;
412  double ang = angle(ri,rj,V.N);
413  ang = (ang >= 0.0) ? ang : 2.0*M_PI+ang;
414  if (ang < min_angle)
415  {
416  min_angle = ang;
417  next_dual = Fj.id;
418  face_index = f;
419  can_add = true;
420  }
421  }
422  }
423  }
424  if (can_add)
425  {
426  V.dual.push_back(next_dual);
427  V.dual_neighbour_map.push_back(V.neigh[face_index]);
428  i++;
429  }
430  if (!can_add && (static_cast<int>(V.dual.size()) != V.n_faces-1))
431  {
432  cerr << endl;
433  cerr << "Unable to order dual vertices of vertex " << V.id << ". There is likely a problem with the mesh itself. ";
434  cerr << "Such problems typically arise if the parameters lead to situations such as a very thin (one-cell-wide) neck of cells or ";
435  cerr << "a part of the system trying to detach from the bulk, which is currently not supported. " << endl;
436  //this->debug_dump("test.off");
437  throw runtime_error("Unable to order vertex dual.");
438  }
439  // for boundary vertices add the hole to the end
440  if (V.boundary && (static_cast<int>(V.dual.size()) == V.n_faces-1))
441  V.dual.push_back(V.faces[V.n_faces-1]);
442  }
443  //Not necessary, faces and neighbours line up closely enough
444  //rotate(V.dual_neighbour_map.begin(), V.dual_neighbour_map.begin()+1, V.dual_neighbour_map.end());
445 
446  // Make sure that dual has the same number of elements as there are faces in the star
447  assert(V.dual.size() == V.faces.size());
448 
449  // And update area
450  this->dual_area(V.id);
451 }

461 double Mesh::dual_area(int v)
462 {
463 
464  Vertex& V = m_vertices[v];
465  if (!V.attached) return 0.0;
466 
467  if (!V.ordered)
468  {
469  cout << V << endl;
470  throw runtime_error("Vertex star has to be ordered before dual area can be computed.");
471  }
472 
473  V.area = 0.0;
474  if (V.dual.size() < 3) return V.area;
475  if (!V.boundary)
476  {
477  for (int f = 0; f < V.n_faces; f++)
478  {
479  int fn = (f == V.n_faces - 1) ? 0 : f + 1;
480  Face& F = m_faces[V.dual[f]];
481  Face& Fn = m_faces[V.dual[fn]];
482  V.area += dot(cross(F.rc,Fn.rc),V.N);
483  }
484  }
485  else
486  {
487  V.area = dot(cross(V.r,m_faces[V.dual[0]].rc),V.N);
488  for (int f = 0; f < V.n_faces-2; f++)
489  {
490  Face& F = m_faces[V.dual[f]];
491  Face& Fn = m_faces[V.dual[f+1]];
492  V.area += dot(cross(F.rc,Fn.rc),V.N);
493  }
494  V.area += dot(cross(m_faces[V.dual[V.n_faces-2]].rc,V.r),V.N);
495  }
496 
497  V.area *= 0.5;
498 
499  return V.area;
