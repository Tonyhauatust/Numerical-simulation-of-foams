33 void ExternalTangentAlign::compute()
34 {
35  int N = m_system->size();
36  double tau = m_tau;
37  int Nbonds = m_system->num_bonds();
38 
39  for (int i = 0; i < Nbonds; i++)
40  {
41  Bond& b = m_system->get_bond(i);
42  Particle& pi = m_system->get_particle(b.i);
43  Particle& pj = m_system->get_particle(b.j);
44  double dx = pj.x - pi.x, dy = pj.y - pi.y, dz = pj.z - pi.z; // tangent along bond between i and j
45  m_system->apply_periodic(dx,dy,dz);
46  double l = sqrt(dx*dx + dy*dy + dz*dz);
47  dx /= l; dy /= l; dz /= l; // normalise tangent vector
48  // Compute torque between director of particle i and tangent
49  double tau_x = pi.ny*dz - pi.nz*dy;
50  double tau_y = pi.nz*dx - pi.nx*dz;
51  double tau_z = pi.nx*dy - pi.ny*dx;
52  if (m_has_params)
53  tau = m_type_params[pi.get_type()-1].tau;
54  pi.tau_x += tau_x/tau;
55  pi.tau_y += tau_y/tau;
56  pi.tau_z += tau_z/tau;
57  // Compute torque between director of particle j and tangent
58  tau_x = pj.ny*dz - pj.nz*dy;
59  tau_y = pj.nz*dx - pj.nx*dz;
60  tau_z = pj.nx*dy - pj.ny*dx;
61  if (m_has_params)
62  tau = m_type_params[pj.get_type()-1].tau;
63  pj.tau_x += tau_x/tau;
64  pj.tau_y += tau_y/tau;
65  pj.tau_z += tau_z/tau;
66  }
67 }
