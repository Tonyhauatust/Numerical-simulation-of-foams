46 void IntegratorActomyo::integrate()
47 {
48  int N = m_system->size();
49  double T = m_temp->get_val(m_system->get_run_step()); // current temperature
50  m_stoch_coeff = sqrt(2.0*T*m_dt/m_zeta);
51  // reset forces and torques
52  m_system->reset_forces();
53  m_system->reset_torques();
54 
55  // compute forces in the current configuration
56  if (m_potential)
57  m_potential->compute(m_dt);
58  // iterate over all particles
59  for (int i = 0; i < N; i++)
60  {
61  Particle& p = m_system->get_particle(i);
62  // If particle is actin particle (type 1) add actin force to it
63  if (p.get_type() == m_actin_type)
64  {
65  double tx, ty, tz;
66  m_system->compute_tangent(p.get_id(),tx,ty,tz);
67  vector<int>& neigh = m_nlist->get_neighbours(p.get_id());
68  double min_dist = 1e7; // Some large number
69  int min_j;
70  bool found = false;
71  for (unsigned int j = 0; j < neigh.size(); j++)
72  {
73  Particle& pj = m_system->get_particle(neigh[j]);
74  double dx = pj.x - p.x, dy = pj.y - p.y, dz = pj.z - p.z;
75  m_system->apply_periodic(dx,dy,dz);
76  double l = sqrt(dx*dx + dy*dy + dz*dz);
77  if (l < m_active_cut) // Only condider myosin beads that are within a cutoff distance
78  {
79  if (l < min_dist && pj.get_type() == m_mysoin_site_type) // Type 3 is active site on myosin
80  {
81  min_dist = l;
82  min_j = pj.get_id(); // id of the closest neighbour
83  found = true;
84  }
85  }
86  }
87  if (found)
88  {
89  Particle& pj = m_system->get_particle(min_j);
90  double dot = pj.fx*tx + pj.fy*ty + pj.fz*tz; // Check the dot product between force on myosin and tangent on actin
91  if (dot >= 0.0) // If it is positive, add force to the actin bead in the direction of the tangent vector
92  {
93  p.fx += 0.5*m_f_active*tx;
94  p.fy += 0.5*m_f_active*ty;
95  p.fz += 0.5*m_f_active*tz;
96  // Also update the myosin bead
97  pj.fx -= 0.5*m_f_active*tx;
98  pj.fy -= 0.5*m_f_active*ty;
99  pj.fz -= 0.5*m_f_active*tz;
100  }
101  }
102  }
103  // Update velocity
104  p.vx = p.fx/m_zeta;
105  p.vy = p.fy/m_zeta;
106  p.vz = p.fz/m_zeta;
107  // Update particle position
108  p.x += m_dt*p.vx + m_stoch_coeff*m_rng->gauss_rng(1.0);
109  p.y += m_dt*p.vy + m_stoch_coeff*m_rng->gauss_rng(1.0);
110  p.z += m_dt*p.vz + m_stoch_coeff*m_rng->gauss_rng(1.0);
111  // Project everything back to the manifold
112  m_constrainer->enforce(p);
113  p.age += m_dt;
114  }
115 }
