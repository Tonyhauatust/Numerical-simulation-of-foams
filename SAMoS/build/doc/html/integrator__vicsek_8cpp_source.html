44 void IntegratorVicsek::integrate()
45 {
46  double noise = m_eta*sqrt(m_dt);
47  int N = m_system->get_group(m_group_name)->get_size();
48  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
49 
50  // reset forces and torques
51  m_system->reset_forces();
52  m_system->reset_torques();
53 
54  // compute forces in the current configuration
55  if (m_potential)
56  m_potential->compute(m_dt);
57  // compute torques in the current configuration
58  if (m_align)
59  m_align->compute();
60  // iterate over all particles
61  for (int i = 0; i < N; i++)
62  {
63  int pi = particles[i];
64  Particle& p = m_system->get_particle(pi);
65  // Update particle velocity
66  p.vx = m_v0*p.tau_x + m_mu*p.fx;
67  p.vy = m_v0*p.tau_y + m_mu*p.fy;
68  p.vz = m_v0*p.tau_z + m_mu*p.fz;
69  // Project everything back to the manifold
70  m_constrainer->enforce(p);
71  // Change orientation of the velocity (in the tangent plane)
72  double theta = 2.0*noise*M_PI*(m_rng->drnd() - 0.5);
73  m_constrainer->rotate_velocity(p,theta);
74  // Update particle position
75  p.x += m_dt*p.vx;
76  p.y += m_dt*p.vy;
77  p.z += m_dt*p.vz;
78  // Project everything back to the manifold
79  m_constrainer->enforce(p);
80  p.age += m_dt;
81  }
82 }
