34 void PairLineTensionPotential::compute(double dt)
35 {
36  double lambda = m_lambda;
37  double l0 = m_l0;
38  int N = m_system->size();
39  Mesh& mesh = m_system->get_mesh();
40 
41  if (m_system->compute_per_particle_energy())
42  {
43  for (int i = 0; i < N; i++)
44  {
45  Particle& p = m_system->get_particle(i);
46  p.set_pot_energy("line_tension",0.0);
47  }
48  }
49 
50  m_potential_energy = 0.0;
51  for (int e = 0; e < mesh.nedges(); e++)
52  {
53  Edge& E = mesh.get_edges()[e];
54  if (E.boundary)
55  {
56  Particle& pi = m_system->get_particle(E.from);
57  Particle& pj = m_system->get_particle(E.to);
58  double dx = pj.x - pi.x, dy = pj.y - pi.y, dz = pj.z - pi.z;
59  m_system->apply_periodic(dx,dy,dz);
60  if (m_has_pair_params)
61  {
62  lambda = m_pair_params[pi.get_type() - 1][pj.get_type() - 1].lambda;
63  l0 = m_pair_params[pi.get_type() - 1][pj.get_type() - 1].l0;
64  }
65  double r_sq = dx*dx + dy*dy + dz*dz;
66  double r = sqrt(r_sq);
67  double dl = r - l0;
68  double pot_eng = 0.5*lambda*dl*dl;
69  m_potential_energy += pot_eng;
70  // Handle force
71  double force_factor = lambda*dl/r;
72  pi.fx += force_factor*dx;
73  pi.fy += force_factor*dy;
74  pi.fz += force_factor*dz;
75  // Use 3d Newton's law
76  pj.fx -= force_factor*dx;
77  pj.fy -= force_factor*dy;
78  pj.fz -= force_factor*dz;
79  if (m_system->compute_per_particle_energy())
80  {
81  pi.add_pot_energy("line_tension",pot_eng);
82  pj.add_pot_energy("line_tension",pot_eng);
83  }
84  }
85  }
86 }
