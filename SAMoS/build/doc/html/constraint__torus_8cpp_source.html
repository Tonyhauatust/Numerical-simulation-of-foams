58 void ConstraintTorus::compute_gradient(Particle&p, double& gx, double& gy, double& gz)
59 {
60  double x = p.x, y = p.y, z = p.z;
61  double sqr_xy = sqrt(x*x + y*y);
62  double fact = -2.0*(m_c - sqr_xy)/sqr_xy;
63  gx = x*fact;
64  gy = y*fact;
65  gz = 2.0*z;
66 }

71 double ConstraintTorus::constraint_value(Particle& p)
72 {
73  double x = p.x, y = p.y, z = p.z;
74  double ft = m_c - sqrt(x*x + y*y);
75  double g = ft*ft + z*z - m_a*m_a; // implicit equation for torus
