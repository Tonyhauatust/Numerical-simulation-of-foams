36 void PairLJRodPotential::compute(double dt)
37 {
38  int N = m_system->size();
39  double sigma = m_sigma;
40  double eps = m_eps;
41  double rcut = m_rcut;
42  double sigma_sq = sigma*sigma, rcut_sq = rcut*rcut;
43  double li2, lj2;
44  double lambda, mu;
45  double dx, dy, dz;
46  double fx, fy, fz;
47  double force_factor;
48  double potential_energy;
49  double alpha_i = 1.0; // phase in factor for particle i
50  double alpha_j = 1.0; // phase in factor for particle j
51  double alpha = 1.0; // phase in factor for pair interaction (see below)
52  double k;
53  double inv_core_sq, inv_core_6, lj_core_sq;
54 
55  if (m_system->compute_per_particle_energy())
56  {
57  for (int i = 0; i < N; i++)
58  {
59  Particle& p = m_system->get_particle(i);
60  p.set_pot_energy("ljrod",0.0);
61  }
62  }
63 
64  m_potential_energy = 0.0;
65  for (int i = 0; i < N; i++)
66  {
67  Particle& pi = m_system->get_particle(i);
68  if (m_phase_in)
69  alpha_i = 0.5*(1.0 + m_val->get_val(static_cast<int>(pi.age/dt)));
70  li2 = 0.5*pi.get_length();
71  double ni_x = pi.nx, ni_y = pi.ny, ni_z = pi.nz;
72  vector<int>& neigh = m_nlist->get_neighbours(i);
73  for (unsigned int j = 0; j < neigh.size(); j++)
74  {
75  Particle& pj = m_system->get_particle(neigh[j]);
76  if (m_phase_in)
77  {
78  alpha_j = 0.5*(1.0 + m_val->get_val(static_cast<int>(pj.age/dt)));
79  // Determine global phase in factor: particles start at 0.5 strength (both daugthers of a division replace the mother)
80  // Except for the interaction between daugthers which starts at 0
81  if (alpha_i<1.0 && alpha_j < 1.0)
82  alpha = alpha_i + alpha_j - 1.0;
83  else
84  alpha = alpha_i*alpha_j;
85  }
86  if (m_has_pair_params)
87  {
88  rcut = m_pair_params[pi.get_type()-1][pj.get_type()-1].rcut;
89  rcut_sq = rcut*rcut;
90  }
91  lj2 = 0.5*pj.get_length();
92  double nj_x = pj.nx, nj_y = pj.ny, nj_z = pj.nz;
93  double dx_cm = pj.x - pi.x, dy_cm = pj.y - pi.y, dz_cm = pj.z - pi.z;
94  m_system->apply_periodic(dx_cm,dy_cm,dz_cm);
95 
96  double ni_dot_nj = ni_x*nj_x + ni_y*nj_y + ni_z*nj_z;
97  double drcm_dot_ni = dx_cm*ni_x + dy_cm*ni_y + dz_cm*ni_z;
98  double drcm_dot_nj = dx_cm*nj_x + dy_cm*nj_y + dz_cm*nj_z;
99  double cc = 1.0 - ni_dot_nj*ni_dot_nj;
100 
101  if (cc <= 1e-10) // rods are nearly parallel
102  {
103  if (fabs(drcm_dot_ni) > 1e-10)
104  {
105  lambda = copysign(li2,drcm_dot_ni);
106  mu = lambda*ni_dot_nj - drcm_dot_nj;
107  if (fabs(mu) > lj2) mu = copysign(lj2,mu);
108  }
109  else
110  {
111  lambda = 0.0;
112  mu = 0.0;
113  }
114 // lambda = 0.5*drcm_dot_ni;
115 // mu = -0.5*drcm_dot_nj;
116 // if (fabs(lambda) > li2) lambda = copysign(li2,lambda);
117 // if (fabs(mu) > lj2) mu = copysign(lj2,mu);
118  }
119  else
120  {
121  lambda = (drcm_dot_ni-ni_dot_nj*drcm_dot_nj)/cc;
122  mu = (-drcm_dot_nj+ni_dot_nj*drcm_dot_ni)/cc;
123  if (fabs(lambda) > li2 || fabs(mu) > lj2)
124  {
125  if (fabs(lambda) - li2 > fabs(mu) - lj2)
126  {
127  lambda = copysign(li2,lambda);
128  mu = lambda*ni_dot_nj - drcm_dot_nj;
129  if (fabs(mu) > lj2) mu = copysign(lj2,mu);
130  }
131  else
132  {
133  mu = copysign(lj2,mu);
134  lambda = mu*ni_dot_nj + drcm_dot_ni;
135  if (fabs(lambda) > li2) lambda = copysign(li2,lambda);
136  }
137  }
138  }
139  dx = dx_cm + mu*nj_x - lambda*ni_x;
140  dy = dy_cm + mu*nj_y - lambda*ni_y;
141  dz = dz_cm + mu*nj_z - lambda*ni_z;
142 
143  double r_sq = dx*dx + dy*dy + dz*dz;
144 
145  if (r_sq <= rcut_sq)
146  {
147  if (m_has_pair_params)
148  {
149  int pi_t = pi.get_type() - 1, pj_t = pj.get_type() - 1;
150  sigma = m_pair_params[pi_t][pj_t].sigma;
151  eps = m_pair_params[pi_t][pj_t].eps;
152  sigma_sq = sigma*sigma;
153  }
154  if (r_sq <= LJ_HARD_CORE_DISTANCE*LJ_HARD_CORE_DISTANCE*sigma_sq)
155  {
156  lj_core_sq = LJ_HARD_CORE_DISTANCE*LJ_HARD_CORE_DISTANCE;
157  inv_core_sq = sigma_sq/lj_core_sq;
158  inv_core_6 = inv_core_sq*inv_core_sq*inv_core_sq;
159  k = 6.0*eps/(lj_core_sq*lj_core_sq)*inv_core_6*(2.0*inv_core_6 - 1.0);
160  potential_energy = 4.0*eps*alpha*inv_core_6*(inv_core_6 - 1.0);
161  if (r_sq <= SMALL_NUMBER)
162  {
163  dx = dx_cm; dy = dy_cm; dz = dz_cm;
164  r_sq = dx_cm*dx_cm + dy_cm*dy_cm + dz_cm*dz_cm;
165  }
166  force_factor = 4.0*k*r_sq*sqrt(r_sq);
167  }
168  else
169  {
170  double inv_r_sq = sigma_sq/r_sq;
171  double inv_r_6 = inv_r_sq*inv_r_sq*inv_r_sq;
172  // Handle potential
173  potential_energy = 4.0*eps*alpha*inv_r_6*(inv_r_6 - 1.0);
174  if (m_shifted)
175  {
176  double inv_r_cut_sq = sigma_sq/rcut_sq;
177  double inv_r_cut_6 = inv_r_cut_sq*inv_r_cut_sq*inv_r_cut_sq;
178  potential_energy -= 4.0 * eps * alpha * inv_r_cut_6 * (inv_r_cut_6 - 1.0);
179  }
180  force_factor = 48.0*eps*alpha*inv_r_6*(inv_r_6 - 0.5)*inv_r_sq;
181  }
182  m_potential_energy += potential_energy;
183  // Handle force
184  fx = force_factor*dx; fy = force_factor*dy; fz = force_factor*dz;
185  pi.fx -= fx;
186  pi.fy -= fy;
187  pi.fz -= fz;
188  // Use 3d Newton's law
189  pj.fx += fx;
190  pj.fy += fy;
191  pj.fz += fz;
192  // handle torques
193  pi.tau_x -= lambda*(ni_y*fz - ni_z*fy);
194  pi.tau_y -= lambda*(ni_z*fx - ni_x*fz);
195  pi.tau_z -= lambda*(ni_x*fy - ni_y*fx);
196  pj.tau_x += mu*(nj_y*fz - nj_z*fy);
197  pj.tau_y += mu*(nj_z*fx - nj_x*fz);
198  pj.tau_z += mu*(nj_x*fy - nj_y*fx);
199  if (m_system->compute_per_particle_energy())
200  {
201  pi.add_pot_energy("ljrod",potential_energy);
202  pj.add_pot_energy("ljrod",potential_energy);
203  }
204  }
205  }
206  }
