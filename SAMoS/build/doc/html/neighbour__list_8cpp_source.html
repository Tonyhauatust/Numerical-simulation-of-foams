32 /* Get neighbour opposite to an edge
33  * \note This is extremely inelegant. However, I cannot figure out
34  * a simple way to tall CGAL directly to get me the other vertex
35  * opposite to an edge.
36 */
37 static int get_opposite(const Delaunay::Face_handle f, const int i, const int j)
38 {
39  int I = f->vertex(0)->info(), J = f->vertex(1)->info(), K = f->vertex(2)->info();
40  if (I == i)
41  {
42  if (J == j) return K;
43  else return J;
44  }
45  else if (J == i)
46  {
47  if (I == j) return K;
48  else return I;
49  }
50  else
51  {
52  if (J == j) return I;
53  else return J;
54  }
55 }

57 // Following two functions are used to determine which trainges are inside the boundary.
58 // These are adopted from: http://doc.cgal.org/latest/Triangulation_2/index.html#title29
59 // Example: Triangulating a Polygonal Domain
60 static void mark_domains(Delaunay& ct, Delaunay::Face_handle start, int index, list<Delaunay::Edge>& border )
61 {
62  if(start->info().nesting_level != -1)
63  {
64  return;
65  }
66  list<Delaunay::Face_handle> queue;
67  queue.push_back(start);
68  while(! queue.empty())
69  {
70  Delaunay::Face_handle fh = queue.front();
71  queue.pop_front();
72  if(fh->info().nesting_level == -1)
73  {
74  fh->info().nesting_level = index;
75  for(int i = 0; i < 3; i++)
76  {
77  Delaunay::Edge e(fh,i);
78  Delaunay::Face_handle n = fh->neighbor(i);
79  if(n->info().nesting_level == -1)
80  {
81  if(ct.is_constrained(e)) border.push_back(e);
82  else queue.push_back(n);
83  }
84  }
85  }
86  }
87 }
88 //explore set of facets connected with non constrained edges,
89 //and attribute to each such set a nesting level.
90 //We start from facets incident to the infinite vertex, with a nesting
91 //level of 0. Then we recursively consider the non-explored facets incident
92 //to constrained edges bounding the former set and increase the nesting level by 1.
93 //Facets in the domain are those with an odd nesting level.
94 void mark_domains(Delaunay& cdt)
95 {
96  for(Delaunay::All_faces_iterator it = cdt.all_faces_begin(); it != cdt.all_faces_end(); ++it)
97  {
98  it->info().nesting_level = -1;
99  }
100  list<Delaunay::Edge> border;
101  mark_domains(cdt, cdt.infinite_face(), 0, border);
102  while(!border.empty())
103  {
104  Delaunay::Edge e = border.front();
105  border.pop_front();
106  Delaunay::Face_handle n = e.first->neighbor(e.second);
107  if(n->info().nesting_level == -1)
108  {
109  mark_domains(cdt, n, e.first->info().nesting_level+1, border);
110  }
111  }
112 }

178 void NeighbourList::build_nsq(int i)
179 {
180  double cut = m_cut+m_pad;
181  double cut2 = cut*cut;
182  double d2;
183 
184  Particle& pi = m_system->get_particle(i);
185  pi.coordination = 0;
186  for (int j = 0; j < pi.get_id(); j++)
187  {
188  Particle& pj = m_system->get_particle(j);
189  bool exclude = false;
190  double dx = pi.x - pj.x;
191  double dy = pi.y - pj.y;
192  double dz = pi.z - pj.z;
193  m_system->apply_periodic(dx,dy,dz);
194  d2 = dx*dx + dy*dy + dz*dz;
195  if (m_system->has_exclusions())
196  if (m_system->in_exclusion(pi.get_id(), pj.get_id()))
197  exclude = true;
198  if (d2 < cut2 && (!exclude))
199  m_list[j].push_back(i);
200  if (d2 < cut2)
201  {
202  double r = pi.get_radius() + pj.get_radius();
203  if (d2 < r*r)
204  {
205  pi.coordination++;
206  pj.coordination++;
207  }
208  }
209 
210  }
211  m_old_state.push_back(PartPos(pi.x,pi.y,pi.z));

278 bool NeighbourList::need_update(Particle& p)
279 {
280  int id = p.get_id();
281  bool periodic = m_system->get_periodic();
282  BoxPtr box = m_system->get_box();
283 
284  double dx = m_old_state[id].x - p.x;
285  double dy = m_old_state[id].y - p.y;
286  double dz = m_old_state[id].z - p.z;
287 
288  if (periodic)
289  {
290  if (dx > box->xhi) dx -= box->Lx;
291  else if (dx < box->xlo) dx += box->Lx;
292  if (dy > box->yhi) dy -= box->Ly;
293  else if (dy < box->ylo) dy += box->Ly;
294  if (dz > box->zhi) dz -= box->Lz;
295  else if (dz < box->zlo) dz += box->Lz;
296  }
297 
298  if (dx*dx + dy*dy + dz*dz < 0.25*m_pad*m_pad)
299  return false;
300  else
301  return true;
302 }

336 bool NeighbourList::build_triangulation()
337 {
338  vector< pair<Point,unsigned> > points;
339  vector< Point > points_for_boundary; // need to double the data structure do the the way CGAL constrained triangulations work
340  vector< pair<int,int> > boundary_indices; // pairs of edges at the boundary
341 
342  int N = m_system->size();
343  vector<int> tissue_id; // Bookkeeping vector for building triangulation
344  vector<int> index_map(N,-1);
345  int tissue_count = 0;
346  for (int i = 0; i < N; i++)
347  {
348  Particle& pi = m_system->get_particle(i);
349  if (pi.in_tissue && pi.z != 0.0)
350  {
351  m_msg->msg(Messenger::ERROR,"Delaunay triangulation is only supported in plane. All z components of tissue particles must be set to zero.");
352  throw runtime_error("Unable to build Delaunay triangulation for non-planar systems.");
353  }
354  if (pi.in_tissue)
355  {
356  //points.push_back( make_pair( Point(pi.x,pi.y), pi.get_id() ) );
357  points.push_back( make_pair( Point(pi.x,pi.y), tissue_count ) );
358  points_for_boundary.push_back(Point(pi.x,pi.y));
359  tissue_id.push_back(pi.get_id());
360  index_map[pi.get_id()] = tissue_count;
361  tissue_count++;
362  }
363  }
364  for (int i = 0; i < N; i++)
365  {
366  Particle& pi = m_system->get_particle(i);
367  if (pi.boundary && pi.in_tissue)
368  {
369  boundary_indices.push_back( make_pair(index_map[pi.get_id()],index_map[pi.boundary_neigh[0]]) );
370  boundary_indices.push_back( make_pair(index_map[pi.get_id()],index_map[pi.boundary_neigh[1]]) );
371  }
372  }
373 
374  Delaunay triangulation;
375  triangulation.insert_constraints(points_for_boundary.begin(), points_for_boundary.end(), boundary_indices.begin(), boundary_indices.end());
376  triangulation.insert(points.begin(),points.end());
377  mark_domains(triangulation);
378 
379  for(Delaunay::Finite_edges_iterator eit = triangulation.finite_edges_begin(); eit != triangulation.finite_edges_end(); eit++)
380  {
381  Delaunay::Face_handle f1 = eit->first;
382  Delaunay::Face_handle f2 = f1->neighbor(eit->second);
383  int i = tissue_id[f1->vertex(f1->cw(eit->second))->info()];
384  int j = tissue_id[f1->vertex(f1->ccw(eit->second))->info()];
385  int k, l;
386  int to_flip;
387  bool can_add = false;
388  bool simple_add = true;
389  // if f1 or f2 are boundary handle this first
390  if (triangulation.is_infinite(f1) || triangulation.is_infinite(f2))
391  {
392  if (triangulation.is_infinite(f2))
393  k = tissue_id[f1->vertex(eit->second)->info()]; // k is the index of the index opposite to the edge (i,j)
394  else
395  k = tissue_id[get_opposite(f2, f1->vertex(f1->cw(eit->second))->info(), f1->vertex(f1->ccw(eit->second))->info())];
396  Particle& pi = m_system->get_particle(i);
397  Particle& pj = m_system->get_particle(j);
398  Particle& pk = m_system->get_particle(k);
399  can_add = !(pi.boundary && pj.boundary && pk.boundary); // We can add this edge only if at least of the vertices is not boundary
400  if (can_add && dot(pk, pi, pj) < 0) // In this case, we know that the Internal vertex in pk, so we compute angle at it
401  {
402  simple_add = false;
403  to_flip = k;
404  }
405  }
406  else
407  {
408  // k is the index of the index opposite to the edge (i,j) for face f1
409  k = tissue_id[f1->vertex(eit->second)->info()];
410  // l is the index of the index opposite to the edge (i,j) for face f2
411  l = tissue_id[get_opposite(f2, f1->vertex(f1->cw(eit->second))->info(), f1->vertex(f1->ccw(eit->second))->info())];
412  Particle& pi = m_system->get_particle(i);
413  Particle& pj = m_system->get_particle(j);
414  Particle& pk = m_system->get_particle(k);
415  Particle& pl = m_system->get_particle(l);
416  bool all_f1_boundary = pi.boundary && pj.boundary && pk.boundary;
417  bool all_f2_boundary = pi.boundary && pj.boundary && pl.boundary;
418  can_add = !(all_f1_boundary && all_f2_boundary) || f1->info().in_domain() || f2->info().in_domain();
419  if (can_add)
420  {
421  if (find(pi.boundary_neigh.begin(),pi.boundary_neigh.end(),pj.get_id()) != pi.boundary_neigh.end()) // Flip only if edge (i,j) is a boundary edge
422  {
423  if ((all_f1_boundary && !all_f2_boundary))
424  if (dot(pl, pi, pj) < 0)
425  {
426  simple_add = false;
427  to_flip = l;
428  }
429  if ((!all_f1_boundary && all_f2_boundary))
430  if (dot(pk, pi, pj) < 0)
431  {
432  simple_add = false;
433  to_flip = k;
434  }
435  }
436  }
437  }
438  if (m_static_boundary) // In case boundary is static, ignore all additions of new vertices
439  simple_add = true;
440  if (can_add)
441  {
442  if (simple_add)
443  {
444  if (find(m_contact_list[i].begin(),m_contact_list[i].end(),j) == m_contact_list[i].end()) m_contact_list[i].push_back(j);
445  if (find(m_contact_list[j].begin(),m_contact_list[j].end(),i) == m_contact_list[j].end()) m_contact_list[j].push_back(i);
446  }
447  else
448  {
449  int i1 = i, i2 = j, i3 = to_flip, i4;
450  Particle& p1 = m_system->get_particle(i1); // boundary particle
451  Particle& p2 = m_system->get_particle(i2); // boundary particle
452  Particle& p3 = m_system->get_particle(i3); // internal particle
453 
454  double x, y, z; // contains coordinates of mirrored particles
455  mirror(p3, p1, p2, x, y, z); // compute position of mirrored particle
456  Particle p(m_system->size(), p1.get_type(), p1.get_radius()); // generate new particle with the "last" id and inhereted type and radius from p1
457  i4 = p.get_id();
458  // set parameters for the new particle
459  p.x = x; p.y = y; p.z = z;
460  p.Nx = p3.Nx; p.Ny = p3.Ny; p.Nz = p3.Nz;
461  p.nx = p3.nx; p.ny = p3.ny; p.nz = p3.nz;
462  p.vx = 0.5*(p1.vx+p2.vx); p.vy = 0.5*(p1.vy+p2.vy); p.vz = 0.5*(p1.vz+p2.vz);
463  p.coordination = 0;
464  // copy all the groups of p1 (we need to do this as internal particles might be in the different group than the boundary)
465  for(list<string>::iterator it_g = p1.groups.begin(); it_g != p1.groups.end(); it_g++)
466  p.add_group(*it_g);
467  // also make sure that the new particle belongs to the boundary group
468  if (m_system->has_group("boundary"))
469  p.add_group("boundary");
470  // Note: Make sure that new particle is added to all necessary groups.
471  p.boundary = true;
472  p.in_tissue = true;
473  if (p1.boundary_neigh.size() == 2) p1.boundary_neigh[(p1.boundary_neigh[0] == i2) ? 0 : 1] = i4;
474  if (p2.boundary_neigh.size() == 2) p2.boundary_neigh[(p2.boundary_neigh[0] == i1) ? 0 : 1] = i4;
475  p.boundary_neigh.push_back(i1);
476  p.boundary_neigh.push_back(i2);
477  p3.boundary = false;
478  p3.boundary_neigh.clear();
479  // add it to the system
480  m_system->add_particle(p);
481  // Extend the contact and neighbour list
482  m_contact_list.push_back(vector<int>());
483  m_list.push_back(vector<int>());
484 
485  // update neighbour list
486  this->build_nsq(p.get_id());
487 
488  if (find(m_contact_list[i3].begin(),m_contact_list[i3].end(),i1) == m_contact_list[i3].end()) m_contact_list[i3].push_back(i1);
489  if (find(m_contact_list[i1].begin(),m_contact_list[i1].end(),i3) == m_contact_list[i1].end()) m_contact_list[i1].push_back(i3);
490 
491  if (find(m_contact_list[i3].begin(),m_contact_list[i3].end(),i2) == m_contact_list[i3].end()) m_contact_list[i3].push_back(i2);
492  if (find(m_contact_list[i2].begin(),m_contact_list[i2].end(),i3) == m_contact_list[i2].end()) m_contact_list[i2].push_back(i3);
493 
494  if (find(m_contact_list[i4].begin(),m_contact_list[i4].end(),i1) == m_contact_list[i4].end()) m_contact_list[i4].push_back(i1);
495  if (find(m_contact_list[i1].begin(),m_contact_list[i1].end(),i4) == m_contact_list[i1].end()) m_contact_list[i1].push_back(i4);
496 
497  if (find(m_contact_list[i4].begin(),m_contact_list[i4].end(),i2) == m_contact_list[i4].end()) m_contact_list[i4].push_back(i2);
498  if (find(m_contact_list[i2].begin(),m_contact_list[i2].end(),i4) == m_contact_list[i2].end()) m_contact_list[i2].push_back(i4);
499 
500  if (find(m_contact_list[i4].begin(),m_contact_list[i4].end(),i3) == m_contact_list[i4].end()) m_contact_list[i4].push_back(i3);
501  if (find(m_contact_list[i3].begin(),m_contact_list[i3].end(),i4) == m_contact_list[i3].end()) m_contact_list[i3].push_back(i4);
502  }
503  }
504 
505  }
506  //this->debug_dump("after_contact_build.mol2");
507 
508  this->remove_dangling();
509  this->remove_detached();
510 
511  //this->debug_dump("after_cleanup.mol2");

513  return true;

558  void NeighbourList::remove_detached()
559  {
560  Mesh& mesh = m_system->get_mesh();
561  if (mesh.size() == 0) return;
562 
563 
564  vector<int> to_remove;
565  int offset = 0; // We need to shift vertex ids to match them in the removal
566  for (unsigned int i = 0; i < m_contact_list.size(); i++)
567  {
568  Particle& pi = m_system->get_particle(i);
569  if (pi.in_tissue && m_contact_list[i].size() == 0)
570  {
571  to_remove.push_back(i-offset);
572  offset++;
573  }
574  }
575 
576  for (unsigned int i = 0; i < to_remove.size(); i++)
577  {
578  m_system->remove_particle(to_remove[i]);
579  if (m_contact_list[to_remove[i]].size() != 0)
580  throw runtime_error("Trying to remove connected particle.");
581  else
582  {
583  m_contact_list.erase(m_contact_list.begin() + to_remove[i]);

618 void mirror(const Particle& p1, const Particle& p2, const Particle& p3, double& x, double& y, double& z)
619 {
620  double x32 = p3.x - p2.x, y32 = p3.y - p2.y, z32 = p3.z - p2.z;
621  double len = sqrt(x32*x32 + y32*y32 + z32*z32);
622  double Nx = x32/len, Ny = y32/len, Nz = z32/len;
623  double n_xx = Nx*Nx, n_xy = Nx*Ny, n_xz = Nx*Nz;
624  double n_yx = Ny*Nx, n_yy = Ny*Ny, n_yz = Ny*Nz;
625  double n_zx = Nz*Nx, n_zy = Nz*Ny, n_zz = Nz*Nz;
626 
627  double P_m_Q_x = p2.x - p1.x, P_m_Q_y = p2.y - p1.y, P_m_Q_z = p2.z - p1.z;
628  x = p1.x + 2.0*((1.0 - n_xx)*P_m_Q_x - n_xy*P_m_Q_y - n_xz*P_m_Q_z);
629  y = p1.y + 2.0*( - n_yx*P_m_Q_x + (1.0 - n_yy)*P_m_Q_y - n_yz*P_m_Q_z);
630  z = p1.z + 2.0*( - n_zx*P_m_Q_x - n_zy*P_m_Q_y + (1.0 - n_zz)*P_m_Q_z);
631 }
