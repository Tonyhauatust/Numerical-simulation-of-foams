32 void BondFenePotential::compute()
33 {
34  int Nbonds = m_system->num_bonds();
35  BoxPtr box = m_system->get_box();
36  double k = m_k;
37  double r0 = m_r0;
38 
39  m_potential_energy = 0.0;
40  for (int i = 0; i < Nbonds; i++)
41  {
42  Bond& b = m_system->get_bond(i);
43  Particle& pi = m_system->get_particle(b.i);
44  Particle& pj = m_system->get_particle(b.j);
45  double dx = pj.x - pi.x, dy = pj.y - pi.y, dz = pj.z - pi.z;
46  m_system->apply_periodic(dx,dy,dz);
47  if (m_has_bond_params)
48  {
49  k = m_bond_params[b.type-1].k;
50  r0 = m_bond_params[b.type-1].r0;
51  }
52 
53  double r_sq = dx*dx + dy*dy + dz*dz;
54  // Handle potential
55  double r0_sq = r0*r0;
56  double fact = 1.0-r_sq/r0_sq;
57  double potential_energy = -0.5*k*r0_sq*log(fact);
58  m_potential_energy += potential_energy;
59  // Handle force
60  double force_factor = -k/fact;
61  pi.fx += force_factor*dx;
62  pi.fy += force_factor*dy;
63  pi.fz += force_factor*dz;
64  // Use 3d Newton's law
65  pj.fx -= force_factor*dx;
66  pj.fy -= force_factor*dy;
67  pj.fz -= force_factor*dz;
68  }
69 }
