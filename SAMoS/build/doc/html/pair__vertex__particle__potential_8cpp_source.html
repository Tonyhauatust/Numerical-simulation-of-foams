48  double K = m_K;
49  double gamma = m_gamma;
50  double lambda = m_lambda;
51  double alpha = 1.0; // phase in factor
52  double pot_eng = 0.0;
53 
54  if (m_mesh_update_steps > 0)
55  if (m_system->get_step() % m_mesh_update_steps == 0)
56  m_nlist->build_mesh();
57 
58  Mesh& mesh = m_system->get_mesh();
59 
60  if (m_system->compute_per_particle_energy())
61  {
62  for (int i = 0; i < N; i++)
63  {
64  Particle& p = m_system->get_particle(i);
65  p.set_pot_energy("vp",0.0);
66  }
67  }
68 
69  m_potential_energy = 0.0;
70  for (int i = 0; i < N; i++)
71  {
72  Particle& pi = m_system->get_particle(i);
73  Vertex& vi = mesh.get_vertices()[i];
74  Vector3d Nvec = Vector3d(pi.Nx, pi.Ny, pi.Nz);
75  if (m_phase_in)
76  alpha = m_val->get_val(static_cast<int>(pi.age/dt));
77  // First handle the vertex itself
78  if (pi.in_tissue && (m_include_boundary || !vi.boundary))
79  {
80  if (m_has_part_params)
81  {
82  K = m_particle_params[vi.type-1].K;
83  gamma = m_particle_params[vi.type-1].gamma;
84  lambda = m_particle_params[vi.type-1].lambda;
85  }
86  double dA = 0.0;
87  dA = (vi.area - pi.A0);
88  double area_term = 0.5*K*dA;
89  double perim_term = gamma*vi.perim;
90  pot_eng = 0.5*(K*dA*dA+gamma*vi.perim*vi.perim);
91  Vector3d area_vec(0.0,0.0,0.0);
92  Vector3d perim_vec(0.0,0.0,0.0);
93  Vector3d con_vec(0.0,0.0,0.0);
94  for (int f = 0; f < vi.n_faces; f++)
95  {
96  int fid = vi.dual[f];
97  int fid_m = vi.dual[prev(f,vi.n_faces)];
98  int fid_p = vi.dual[next(f,vi.n_faces)];
99 
100  Face& f_nu_m = mesh.get_faces()[fid_m];
101  Face& f_nu = mesh.get_faces()[fid];
102  Face& f_nu_p = mesh.get_faces()[fid_p];
103 
104  Vector3d& r_nu_m = f_nu_m.rc;
105  Vector3d& r_nu = f_nu.rc;
106  Vector3d& r_nu_p = f_nu_p.rc;
107 
108  Vector3d cross_prod_1(0.0,0.0,0.0);
109  if (!(f_nu.is_hole || f_nu_p.is_hole)) cross_prod_1 = (cross(r_nu_p, Nvec))*f_nu.get_jacobian(i);
110  if (!(f_nu_m.is_hole || f_nu.is_hole)) cross_prod_1 = cross_prod_1 - (cross(r_nu_m, Nvec))*f_nu.get_jacobian(i);
111  area_vec = area_vec + cross_prod_1;
112 
113  Vector3d cross_prod_2(0.0,0.0,0.0);
114  if (!(f_nu_m.is_hole || f_nu.is_hole)) cross_prod_2 = ((r_nu - r_nu_m).unit())*f_nu.get_jacobian(i);
115  if (!(f_nu_p.is_hole || f_nu.is_hole)) cross_prod_2 -= (((r_nu_p - r_nu).unit())*f_nu.get_jacobian(i));
116  perim_vec = perim_vec + cross_prod_2;
117 
118  // Adding force contributions from the dual edges associated with face f and face next(f,vi.n_faces)
119  if (m_has_pair_params)
120  {
121  Vertex& vn = mesh.get_vertices()[vi.dual_neighbour_map[f]];
122  lambda = m_pair_params[vi.type-1][vn.type-1].lambda;
123  }
124  Vector3d cross_prod_3(0.0,0.0,0.0);
125  if (!(f_nu_m.is_hole || f_nu.is_hole)) cross_prod_3 = lambda*(((r_nu - r_nu_m).unit())*f_nu.get_jacobian(i));
126  con_vec = con_vec + cross_prod_3;
127 
128  // Adding force contributions from the dual edges associated with face f and face prev(f,vi.n_faces)
129  if (m_has_pair_params)
130  {
131  Vertex& vn = mesh.get_vertices()[vi.dual_neighbour_map[next(f,vi.n_faces)]];
132  lambda = m_pair_params[vi.type-1][vn.type-1].lambda;
133  }
134  Vector3d cross_prod_4(0.0,0.0,0.0);
135  if (!(f_nu_p.is_hole || f_nu.is_hole)) cross_prod_4 = lambda*(((r_nu_p - r_nu).unit())*f_nu.get_jacobian(i));
136  con_vec = con_vec - cross_prod_4;
137 
138  pot_eng += lambda*(r_nu - r_nu_m).len();
139  }
140  // area term
141  double area_fact = -alpha*area_term;
142  pi.fx += area_fact*area_vec.x;
143  pi.fy += area_fact*area_vec.y;
144  pi.fz += area_fact*area_vec.z;
145  // perimeter term
146  double perim_fact = -alpha*perim_term;
147  pi.fx += perim_fact*perim_vec.x;
148  pi.fy += perim_fact*perim_vec.y;
149  pi.fz += perim_fact*perim_vec.z;
150  // Add contractile term
151  pi.fx -= alpha*con_vec.x;
152  pi.fy -= alpha*con_vec.y;
153  pi.fz -= alpha*con_vec.z;
154  // add potential energy
155  m_potential_energy += pot_eng;
156  }
157  // Now check neighbours
158  for (int j = 0; j < vi.n_edges; j++)
159  {
160  Particle& pj = m_system->get_particle(vi.neigh[j]);
161  Vertex& vj = mesh.get_vertices()[vi.neigh[j]];
162  if (pj.in_tissue && (m_include_boundary || !vj.boundary))
163  {
164  if (m_has_part_params)
165  {
166  K = m_particle_params[vj.type-1].K;
167  gamma = m_particle_params[vj.type-1].gamma;
168  lambda = m_particle_params[vj.type-1].lambda;
169  }
170  double dA = 0.0;
171  dA = (vj.area - pj.A0);
172  double area_term = 0.5*K*dA;
173  double perim_term = gamma*vj.perim;
174  Vector3d area_vec(0.0,0.0,0.0);
175  Vector3d perim_vec(0.0,0.0,0.0);
176  Vector3d con_vec(0.0,0.0,0.0);
177  Vector3d Nvec = Vector3d(pj.Nx, pj.Ny, pj.Nz);
178 
179  for (int f = 0; f < vj.n_faces; f++)
180  {
181  int fid = vj.dual[f];
182  Face& f_nu = mesh.get_faces()[fid];
183  if (f_nu.has_vertex(i))
184  {
185  int fid_m = vj.dual[prev(f,vj.n_faces)];
186  int fid_p = vj.dual[next(f,vj.n_faces)];
187  Face& f_nu_m = mesh.get_faces()[fid_m];
188  Face& f_nu_p = mesh.get_faces()[fid_p];
189  Vector3d& r_nu_m = f_nu_m.rc;
190  Vector3d& r_nu = f_nu.rc;
191  Vector3d& r_nu_p = f_nu_p.rc;
192 
193  Vector3d cross_prod_1(0.0,0.0,0.0);
194  if (!(f_nu.is_hole || f_nu_p.is_hole)) cross_prod_1 = cross(r_nu_p, Nvec)*f_nu.get_jacobian(i);
195  if (!(f_nu_m.is_hole || f_nu.is_hole)) cross_prod_1 = cross_prod_1 - cross(r_nu_m, Nvec)*f_nu.get_jacobian(i);
196  area_vec = area_vec + cross_prod_1;
197 
198  Vector3d cross_prod_2(0.0,0.0,0.0);
199  if (!(f_nu_m.is_hole || f_nu.is_hole)) cross_prod_2 = ((r_nu - r_nu_m).unit())*f_nu.get_jacobian(i);
200  if (!(f_nu_p.is_hole || f_nu.is_hole)) cross_prod_2 -= ((r_nu_p - r_nu).unit())*f_nu.get_jacobian(i);
201  perim_vec = perim_vec + cross_prod_2;

207  lambda = m_pair_params[vj.type-1][vn.type-1].lambda;
208  }

210  if (!(f_nu_m.is_hole || f_nu.is_hole)) cross_prod_3 = lambda*(((r_nu - r_nu_m).unit())*f_nu.get_jacobian(i));
211  con_vec = con_vec + cross_prod_3;
212 
213  // Adding force contributions from the dual edges associated with face f and face next(f,vi.n_faces)
214  if (m_has_pair_params)
215  {
216  Vertex& vn = mesh.get_vertices()[vj.dual_neighbour_map[next(f,vj.n_faces)]];
217  lambda = m_pair_params[vj.type-1][vn.type-1].lambda;
218  }
219  Vector3d cross_prod_4(0.0,0.0,0.0);
220  if (!(f_nu_p.is_hole || f_nu.is_hole)) cross_prod_4 = lambda*(((r_nu_p - r_nu).unit())*f_nu.get_jacobian(i));
221  con_vec = con_vec - cross_prod_4;
222 
223  }
224  }

226  double area_fact = -alpha*area_term;

228  pi.fy += area_fact*area_vec.y;

230  // perimeter term

232  pi.fx += perim_fact*perim_vec.x;
233  pi.fy += perim_fact*perim_vec.y;

238  pi.fz -= alpha*con_vec.z;

242  {

246  pi.s_zx *= inv_area; pi.s_zy *= inv_area; pi.s_zz *= inv_area;
