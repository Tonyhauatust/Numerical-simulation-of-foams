36 void IntegratorBrownianRod::integrate()
37 {
38  int N = m_system->get_group(m_group_name)->get_size();
39  //double T = m_temp->get_val(m_system->get_run_step());
40  double fd_x, fd_y, fd_z; // Deterministic part of the force
41  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
42  double R1, R2;
43 
44  // reset forces and torques
45  m_system->reset_forces();
46  m_system->reset_torques();
47 
48  // If nematic, attempt to flip directors
49  if (m_nematic)
50  for (int i = 0; i < N; i++)
51  {
52  int pi = particles[i];
53  Particle& p = m_system->get_particle(pi);
54  if (m_rng->drnd() < m_tau) // Flip direction n with probability m_tua (dt/tau, where tau is the parameter given in the input file).
55  {
56  p.nx = -p.nx; p.ny = -p.ny; p.nz = -p.nz;
57  }
58  }
59 
60  // compute forces in the current configuration
61  if (m_potential)
62  m_potential->compute(m_dt);
63  // compute torques in the current configuration
64  if (m_align)
65  m_align->compute();
66  // iterate over all particles
67  for (int i = 0; i < N; i++)
68  {
69  int pi = particles[i];
70  Particle& p = m_system->get_particle(pi);
71  double nx = p.nx, ny = p.ny, nz = p.nz;
72  double P = 0.5*p.get_length()/p.get_radius(); // aspect ration defined as length over diameter
73  double D_perp = m_D0/(4.0*M_PI)*(log(P) + 0.839 + 0.185/P + 0.233/(P*P));
74  double D_par = m_D0/(2.0*M_PI)*(log(P) - 0.207 + 0.980/P - 0.133/(P*P));
75  double D_rot = 3.0*m_D0/(M_PI*p.get_length())*(log(P) - 0.662 + 0.917/P - 0.050/(P*P));
76 
77  // Project force onto director
78  double f_dot_n = p.fx*nx + p.fy*ny + p.fz*nz;
79  double fx_par = f_dot_n*nx, fy_par = f_dot_n*ny, fz_par = f_dot_n*nz;
80  double fx_perp = p.fx - fx_par, fy_perp = p.fy - fy_par, fz_perp = p.fz - fz_par;
81  // compute deterministic forces
82  fd_x = m_v0*p.nx + D_par*fx_par + D_perp*fx_perp;
83  fd_y = m_v0*p.ny + D_par*fy_par + D_perp*fy_perp;
84  fd_z = m_v0*p.nz + D_par*fz_par + D_perp*fz_perp;
85  // Update velocity
86  p.vx = fd_x;
87  p.vy = fd_y;
88  p.vz = fd_z;
89  // Update particle position according to the eq. (1a)
90  p.x += m_dt*fd_x;
91  p.y += m_dt*fd_y;
92  p.z += m_dt*fd_z;
93 
94  // Normal to the manifold
95  double w1_x, w1_y, w1_z;
96 
97  m_constrainer->compute_normal(p,w1_x,w1_y,w1_z);
98 
99  // normal to both director and the manifold normal, i.e., the cotangent vector
100  double w2_x = ny*w1_z - nz*w1_y;
101  double w2_y = nz*w1_x - nx*w1_z;
102  double w2_z = nx*w1_y - ny*w1_x;
103 
104  if (m_pos_noise)
105  {
106  double stoch_par = sqrt(2.0*D_par*m_dt);
107  double stoch_perp = sqrt(2.0*D_perp*m_dt);
108  R1 = m_rng->gauss_rng(1.0);
109  R2 = m_rng->gauss_rng(1.0);
110 
111  p.x += stoch_par*m_rng->gauss_rng(1.0)*nx + stoch_perp*(R1*w1_x + R2*w2_x);
112  p.y += stoch_par*m_rng->gauss_rng(1.0)*ny + stoch_perp*(R1*w1_y + R2*w2_y);
113  p.z += stoch_par*m_rng->gauss_rng(1.0)*nz + stoch_perp*(R1*w1_z + R2*w2_z);
114  }
115 
116  double stoch = sqrt(2.0*D_rot*m_dt);
117  R1 = m_rng->gauss_rng(1.0);
118  R2 = m_rng->gauss_rng(1.0);
119 
120  double dnx = p.tau_y*nz - p.tau_z*ny;
121  double dny = p.tau_z*nx - p.tau_x*nz;
122  double dnz = p.tau_x*ny - p.tau_y*nx;
123 
124  p.nx += D_rot*dnx*m_dt + stoch*(R1*w1_x + R2*w2_x);
125  p.ny += D_rot*dny*m_dt + stoch*(R1*w1_y + R2*w2_y);
126  p.nz += D_rot*dnz*m_dt + stoch*(R1*w1_z + R2*w2_z);
127 
128 
129  // Project everything back to the manifold
130  m_constrainer->enforce(p);
131  // Compute angular velocity
132  p.omega += m_dt*m_constrainer->project_torque(p);
133  // Update rod age
134  p.age += m_dt;
135  }
136 }
