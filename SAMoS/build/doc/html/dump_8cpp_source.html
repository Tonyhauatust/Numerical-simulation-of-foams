419 void Dump::dump_dcd()
420 {
421  double Lx = m_system->get_box()->Lx;
422  double Ly = m_system->get_box()->Ly;
423  double Lz = m_system->get_box()->Lz;
424  int N = m_system->size();
425  float* buffer = new float[N]; // buffer for printing out coordinates; note: DCD requires floats
426  double unitcell[6];
427  // set box dimensions (not really important
428  unitcell[0] = 100.0f;
429  unitcell[2] = 100.0f;
430  unitcell[5] = 100.0f;
431  // box angles are 90 degrees
432  unitcell[1] = 0.0f;
433  unitcell[3] = 0.0f;
434  unitcell[4] = 0.0f;
435  write_int(m_file, 48);
436  m_file.write((char *)unitcell, 48);
437  write_int(m_file, 48);
438  // check for errors
439  if (!m_file.good())
440  {
441  m_msg->msg(Messenger::ERROR,"Error writing into DCD file.");
442  throw runtime_error("Error writing DCD file");
443  }
444  // Prepare x coordinate
445  for (int i = 0; i < N; i++)
446  {
447  Particle& p = m_system->get_particle(i);
448  if (m_params.find("unwrap") != m_params.end())
449  buffer[i] = static_cast<float>(p.x+p.ix*Lx);
450  else
451  buffer[i] = static_cast<float>(p.x);
452  }
453  // write x coords
454  write_int(m_file, N * sizeof(float));
455  m_file.write((char *) buffer, N * sizeof(float));
456  write_int(m_file, N * sizeof(float));
457 
458  // Prepare y coordinate
459  for (int i = 0; i < N; i++)
460  {
461  Particle& p = m_system->get_particle(i);
462  if (m_params.find("unwrap") != m_params.end())
463  buffer[i] = static_cast<float>(p.y+p.iy*Ly);
464  else
465  buffer[i] = static_cast<float>(p.y);
466  }
467  // write y coords
468  write_int(m_file, N * sizeof(float));
469  m_file.write((char *) buffer, N * sizeof(float));
470  write_int(m_file, N * sizeof(float));
471 
472  // Prepare z coordinate
473  for (int i = 0; i < N; i++)
474  {
475  Particle& p = m_system->get_particle(i);
476  if (m_params.find("unwrap") != m_params.end())
477  buffer[i] = static_cast<float>(p.z+p.iz*Lz);
478  else
479  buffer[i] = static_cast<float>(p.z);
480  }
481  // write z coords
482  write_int(m_file, N * sizeof(float));
483  m_file.write((char *) buffer, N * sizeof(float));
484  write_int(m_file, N * sizeof(float));

848  }
849  if (rcut > m_nlist->get_cutoff())
850  {
851  m_msg->msg(Messenger::WARNING,"Contact network cutoff distance larger than the neighbour list cutoff. Setting it to that of the neighbour list.");
852  rcut = m_nlist->get_cutoff();
