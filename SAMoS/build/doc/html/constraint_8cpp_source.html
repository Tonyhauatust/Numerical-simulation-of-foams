40 void Constraint::enforce(Particle& p)
41 {
42  bool apply = false;
43  if (m_group == "all")
44  apply = true;
45  else
46  apply = (find(p.groups.begin(),p.groups.end(),m_group) != p.groups.end());
47  if (apply)
48  {
49  // Compute reference gradient (SHAKE method)
50  double ref_gx, ref_gy, ref_gz;
51  this->compute_gradient(p,ref_gx, ref_gy, ref_gz);
52 
53  bool satisfied;
54  int iter = 0;
55  while (iter++ < m_max_iter)
56  {
57  satisfied = true;
58  // Compute unit gradient of the implicit function
59  double gx, gy, gz;
60  this->compute_gradient(p,gx,gy,gz);
61  double s = gx*ref_gx + gy*ref_gy + gz*ref_gz;
62  double g = this->constraint_value(p);
63  double lambda = g/s;
64  if (fabs(g) > m_tol)
65  satisfied = false;
66  if (satisfied) break;
67  p.x -= lambda*ref_gx;
68  p.y -= lambda*ref_gy;
69  p.z -= lambda*ref_gz;
70  }
71 
72  double Nx, Ny, Nz;
73  this->compute_normal(p,Nx,Ny,Nz);
74  // compute v.N
75  double v_dot_N = p.vx*Nx + p.vy*Ny + p.vz*Nz;
76  // compute n.N
77  double n_dot_N = p.nx*Nx + p.ny*Ny + p.nz*Nz;
78  // Project velocity onto tangent plane
79  p.vx -= v_dot_N*Nx; p.vy -= v_dot_N*Ny; p.vz -= v_dot_N*Nz;
80  // Project director onto tangent plane
81  p.nx -= n_dot_N*Nx; p.ny -= n_dot_N*Ny; p.nz -= n_dot_N*Nz;
82  // normalize director
83  double inv_len = 1.0/sqrt(p.nx*p.nx + p.ny*p.ny + p.nz*p.nz);
84  p.nx *= inv_len; p.ny *= inv_len; p.nz *= inv_len;
85  m_system->enforce_periodic(p);
86  }
87 }

96 void Constraint::rotate_director(Particle& p, double phi)
97 {
98  bool apply = false;
99  if (m_group == "all")
100  apply = true;
101  else
102  apply = (find(p.groups.begin(),p.groups.end(),m_group) != p.groups.end());
103  if (apply)
104  {
105  double U, V, W;
106  this->compute_normal(p,U,V,W);
107  // Compute angle sins and cosines
108  double c = cos(phi), s = sin(phi);
109  // Compute new velocity coordinates
110  double nx = U*(U*p.nx+V*p.ny+W*p.nz)*(1.0-c) + p.nx*c + (-W*p.ny+V*p.nz)*s;
111  double ny = V*(U*p.nx+V*p.ny+W*p.nz)*(1.0-c) + p.ny*c + ( W*p.nx-U*p.nz)*s;
112  double nz = W*(U*p.nx+V*p.ny+W*p.nz)*(1.0-c) + p.nz*c + (-V*p.nx+U*p.ny)*s;
113  double len = sqrt(nx*nx + ny*ny + nz*nz);
114  // Update particle director (normalize it along the way to collect for any numerical drift that may have occurred)
115  p.nx = nx/len;
116  p.ny = ny/len;
117  p.nz = nz/len;
118  }
119 }
