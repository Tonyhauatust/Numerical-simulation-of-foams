33 void PairSoftAttractivePotential::compute(double dt)
34 {
35  int N = m_system->size();
36  double k = m_k;
37  double ai, aj;
38  double force_factor = 0.0;
39  double alpha_i = 1.0; // phase in factor for particle i
40  double alpha_j = 1.0; // phase in factor for particle j
41  double alpha = 1.0; // phase in factor for pair interaction (see below)
42 
43  for (int i = 0; i < N; i++)
44  {
45  Particle& p = m_system->get_particle(i);
46  if (m_system->compute_per_particle_energy())
47  {
48  p.set_pot_energy("soft_attractive",0.0);
49  }
50  // currently just hacking (for consistency with Daniel's code): Coordination number is interaction number
51  // Reset to 0 here and then recalculation in interaction below
52  p.coordination=0;
53  }
54 
55  m_potential_energy = 0.0;
56  for (int i = 0; i < N; i++)
57  {
58  Particle& pi = m_system->get_particle(i);
59  // A note on phasing in: m_val, the value object, has been pre-set with the number of phase in time steps
60  // It will give a linear interpolation between 0 and 1 based on the current age of the particle in time steps
61  // Second note: I could set the interpolation between 0.5 and 1, however then pair_vertex potential would be dubious
62  // Instead this is done manually here
63  if (m_phase_in)
64  alpha_i = 0.5*(1.0 + m_val->get_val(static_cast<int>(pi.age/dt)));
65  ai = pi.get_radius();
66  vector<int>& neigh = m_nlist->get_neighbours(i);
67  for (unsigned int j = 0; j < neigh.size(); j++)
68  {
69  Particle& pj = m_system->get_particle(neigh[j]);
70  if (m_phase_in)
71  {
72  alpha_j = 0.5*(1.0 + m_val->get_val(static_cast<int>(pj.age/dt)));
73  // Determine global phase in factor: particles start at 0.5 strength (both daugthers of a division replace the mother)
74  // Except for the interaction between daugthers which starts at 0
75  if (alpha_i < 1.0 && alpha_j < 1.0)
76  alpha = alpha_i + alpha_j - 1.0;
77  else
78  alpha = alpha_i*alpha_j;
79  }
80  k = m_pair_params[pi.get_type()-1][pj.get_type()-1].k;
81  double fact = m_pair_params[pi.get_type()-1][pj.get_type()-1].fact;
82  aj = pj.get_radius();
83  double dx = pj.x - pi.x, dy = pj.y - pi.y, dz = pj.z - pi.z;
84  m_system->apply_periodic(dx,dy,dz);
85  double r_sq = dx*dx + dy*dy + dz*dz;
86  double r = sqrt(r_sq);
87  double ai_p_aj;
88  if (!m_use_particle_radii)
89  ai_p_aj = m_pair_params[pi.get_type()-1][pj.get_type()-1].a;
90  else
91  ai_p_aj = ai+aj;
92  double r_turn = fact*ai_p_aj;
93  double r_f = 2.0*r_turn - ai_p_aj;
94  double diff;
95  double pot_eng = 0.0;
96  if (r < r_f)
97  {
98  pi.coordination+=1;
99  pj.coordination+=1;
100  if (r <= r_turn)
101  {
102  diff = ai_p_aj - r;
103  pot_eng = 0.5*k*alpha*diff*diff;
104  if (r > 0.0) force_factor = k*alpha*diff/r;
105  else force_factor = k*diff;
106  }
107  else if (r > r_turn && r < r_f)
108  {
109  diff = r_f - r;
110  pot_eng = -0.5*k*alpha*diff*diff;
111  force_factor = -k*alpha*diff/r;
112  }
113  m_potential_energy += pot_eng;
114  // Handle force
115  pi.fx -= force_factor*dx;
116  pi.fy -= force_factor*dy;
117  pi.fz -= force_factor*dz;
118  // Use 3d Newton's law
119  pj.fx += force_factor*dx;
120  pj.fy += force_factor*dy;
121  pj.fz += force_factor*dz;
122  if (m_system->compute_per_particle_energy())
123  {
124  pi.add_pot_energy("soft_attractive",pot_eng);
125  pj.add_pot_energy("soft_attractive",pot_eng);
126  }
127  }
128  }
129  }
130 }
