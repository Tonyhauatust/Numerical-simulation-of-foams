35 {

38  double wc = m_wc;
39  int N = m_system->size();
40  Mesh& mesh = m_system->get_mesh();
41 
42  if (m_system->compute_per_particle_energy())
43  {
44  for (int i = 0; i < N; i++)
45  {
46  Particle& p = m_system->get_particle(i);
47  p.set_pot_energy("boundary_attraction",0.0);
48  }
49  }
50 
51  m_potential_energy = 0.0;
52  for (int v = 0; v < mesh.size(); v++)
53  {
54  Vertex& Vi = mesh.get_vertices()[v];
55  if (Vi.boundary)
56  {
57  Particle& pi = m_system->get_particle(Vi.id);
58  for (int j = 0; j < Vi.n_edges; j++)
59  {
60  Vertex& Vj = mesh.get_vertices()[Vi.neigh[j]];
61  if (!(Vj.boundary && m_exclude_boundary))
62  {
63  Particle& pj = m_system->get_particle(Vj.id);
64  if (m_has_pair_params)
65  {
66  epsilon = m_pair_params[pi.get_type()-1][pj.get_type()-1].epsilon;
67  rc = m_pair_params[pi.get_type()-1][pj.get_type()-1].rc;
68  wc = m_pair_params[pi.get_type()-1][pj.get_type()-1].wc;
69  }
70  double dx = pi.x - pj.x, dy = pi.y - pj.y, dz = pi.z - pj.z;
71  m_system->apply_periodic(dx,dy,dz);
72  double r = sqrt(dx*dx + dy*dy + dz*dz);
73  if (r >= rc && r <= (rc+wc))
74  {
75  double cos_fac = cos(0.5*M_PI*(r-rc)/wc);
76  double potential_energy = -epsilon*cos_fac*cos_fac;
77  m_potential_energy += potential_energy;
78  double fact = -0.5*M_PI*epsilon/wc*sin(M_PI*(r-rc)/wc)/r;
79  pi.fx += fact*dx;
80  pi.fy += fact*dy;
81  pi.fz += fact*dz;
82  pj.fx -= fact*dx;

84  pj.fz -= fact*dz;
85  if (m_system->compute_per_particle_energy())
86  {
87  pi.add_pot_energy("boundary_attraction",potential_energy);
