54 void IntegratorLangevin::integrate_baoab()
55 {
56  int N = m_system->get_group(m_group_name)->get_size();
57  double T = m_temp->get_val(m_system->get_run_step());
58  double B = sqrt(T*(1.0-exp(-2.0*m_gamma*m_dt)));
59  double exp_dt = exp(-m_gamma*m_dt);
60  double dt2 = 0.5*m_dt;
61  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
62 
63  // BAOA steps
64  for (int i = 0; i < N; i++)
65  {
66  int pi = particles[i];
67  Particle& p = m_system->get_particle(pi);
68  double fact = dt2/p.mass;
69  // B step
70  p.vx += fact*p.fx;
71  p.vy += fact*p.fy;
72  p.vz += fact*p.fz;
73  // A step
74  p.x += dt2*p.vx;
75  p.y += dt2*p.vy;
76  p.z += dt2*p.vz;
77  // Project everything back to the manifold
78  m_constrainer->enforce(p);
79  // O step
80  p.vx *= exp_dt;
81  p.vy *= exp_dt;
82  p.vz *= exp_dt;
83  if (B != 0.0)
84  {
85  double stoch_fact = B/sqrt(p.mass);
86  p.vx += stoch_fact*m_rng->gauss_rng(1.0);
87  p.vy += stoch_fact*m_rng->gauss_rng(1.0);
88  p.vz += stoch_fact*m_rng->gauss_rng(1.0);
89  }
90  // A step
91  p.x += dt2*p.vx;
92  p.y += dt2*p.vy;
93  p.z += dt2*p.vz;
94  // Project everything back to the manifold
95  m_constrainer->enforce(p);
96  }
97 
98  // reset forces
99  m_system->reset_forces();
100 
101  // compute forces in the current configuration
102  if (m_potential)
103  m_potential->compute(m_dt);
104 
105  // B step
106  for (int i = 0; i < N; i++)
107  {
108  int pi = particles[i];
109  Particle& p = m_system->get_particle(pi);
110  double fact = dt2/p.mass;
111  p.vx += fact*p.fx;
112  p.vy += fact*p.fy;
113  p.vz += fact*p.fz;
114  p.age += m_dt;
115  }
116  // Update vertex mesh
117  m_system->update_mesh();
118 }

128 void IntegratorLangevin::integrate_spv()
129 {
130  int N = m_system->get_group(m_group_name)->get_size();
131  double T = m_temp->get_val(m_system->get_run_step());
132  double zeta = sqrt(T*(1.0-exp(-2.0*m_gamma*m_dt)));
133  double eta;
134  if (m_gamma == 0.0) eta = 0.0;
135  else eta = (1.0-exp(-m_dt*m_gamma))/m_gamma;
136  double exp_dt = exp(-m_dt*m_gamma);
137  double dt2 = 0.5*m_dt;
138  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
139 
140  // Step 1
141  for (int i = 0; i < N; i++)
142  {
143  int pi = particles[i];
144  Particle& p = m_system->get_particle(pi);
145  p.x += dt2*p.vx;
146  p.y += dt2*p.vy;
147  p.z += dt2*p.vz;
148  // Project everything back to the manifold
149  m_constrainer->enforce(p);
150  }
151 
152  // reset forces
153  m_system->reset_forces();
154 
155  // compute forces in the current configuration
156  if (m_potential)
157  m_potential->compute(m_dt);
158 
159  // Steps 2 and 3
160  for (int i = 0; i < N; i++)
161  {
162  int pi = particles[i];
163  Particle& p = m_system->get_particle(pi);
164  double fact = eta/p.mass;
165  p.vx = exp_dt*p.vx + fact*p.fx;
166  p.vy = exp_dt*p.vy + fact*p.fy;
167  p.vz = exp_dt*p.vz + fact*p.fz;
168  if (zeta != 0.0)
169  {
170  double stoch_fact = zeta/sqrt(p.mass);
171  p.vx += stoch_fact*m_rng->gauss_rng(1.0);
172  p.vy += stoch_fact*m_rng->gauss_rng(1.0);
173  p.vz += stoch_fact*m_rng->gauss_rng(1.0);
174  }
175  // Step 3
176  p.x += dt2*p.vx;
177  p.y += dt2*p.vy;
178  p.z += dt2*p.vz;
179  // Project everything back to the manifold
180  m_constrainer->enforce(p);
181  p.age += m_dt;
182  }

193 void IntegratorLangevin::integrate_bbk()
194 {
195  int N = m_system->get_group(m_group_name)->get_size();
196  double T = m_temp->get_val(m_system->get_run_step());
197  double B = sqrt(2.0*T*m_dt*m_gamma);
198  double dt2 = 0.5*m_dt;
199  double one_m_dt2 = 1.0 - m_gamma*dt2;
200  double one_div_one_p_dt2 = 1.0/(1.0 + m_gamma*dt2);
201  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
202 
203  // Steps 1 and 2
204  for (int i = 0; i < N; i++)
205  {
206  int pi = particles[i];
207  Particle& p = m_system->get_particle(pi);
208  double dt2_over_mass = dt2/p.mass;
209  p.vx = one_m_dt2*p.vx + dt2_over_mass*p.fx;
210  p.vy = one_m_dt2*p.vy + dt2_over_mass*p.fy;
211  p.vz = one_m_dt2*p.vz + dt2_over_mass*p.fz;
212  if (B != 0.0)
213  {
214  double stoch_fact = 0.5*B/sqrt(p.mass);
215  p.vx += stoch_fact*m_Rx[i];
216  p.vy += stoch_fact*m_Ry[i];
217  p.vz += stoch_fact*m_Rz[i];
218  }
219  p.x += m_dt*p.vx;
220  p.y += m_dt*p.vy;
221  p.z += m_dt*p.vz;
222  // Project everything back to the manifold
223  m_constrainer->enforce(p);
224  }
225 
226  // reset forces
227  m_system->reset_forces();
228 
229  // compute forces in the current configuration
230  if (m_potential)
231  m_potential->compute(m_dt);
232 
233  for (int i = 0; i < N; i++)
234  {
235  int pi = particles[i];
236  Particle& p = m_system->get_particle(pi);
237  double dt2_over_mass = dt2/p.mass;
238  p.vx = one_div_one_p_dt2*(p.vx + dt2_over_mass*p.fx);
239  p.vy = one_div_one_p_dt2*(p.vy + dt2_over_mass*p.fy);
240  p.vz = one_div_one_p_dt2*(p.vz + dt2_over_mass*p.fz);
241  if (B != 0.0)
242  {
243  double stoch_fact = 0.5*B*one_div_one_p_dt2/sqrt(p.mass);
244  m_Rx[i] = m_rng->gauss_rng(1.0);
245  m_Ry[i] = m_rng->gauss_rng(1.0);
246  m_Rz[i] = m_rng->gauss_rng(1.0);
247  p.vx += stoch_fact*m_Rx[i];
248  p.vy += stoch_fact*m_Ry[i];
249  p.vz += stoch_fact*m_Rz[i];
250  p.age += m_dt;
251  }
252  }
