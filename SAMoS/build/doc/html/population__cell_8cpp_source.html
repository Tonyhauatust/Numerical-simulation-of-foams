77 void PopulationCell::divide(int t)
78 {
79  if (m_freq > 0 && t % m_freq == 0) // Attempt division only at certain time steps
80  {
81  if (!m_system->group_ok(m_group_name))
82  {
83  cout << "Before cell division: Group info mismatch for group : " << m_group_name << endl;
84  throw runtime_error("Group mismatch.");
85  }
86  double fact = m_freq*m_div_rate*m_system->get_integrator_step();
87  Mesh& mesh = m_system->get_mesh();
88  int N = m_system->get_group(m_group_name)->get_size();
89  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
90  for (int i = 0; i < N; i++)
91  {
92  int pi = particles[i];
93  Particle& p = m_system->get_particle(pi);
94  Vertex& V = mesh.get_vertices()[p.get_id()];
95  // actual probability of dividing now: (attempt_freq * dt) * m_div_rate * (V.area - max_area)
96  // here m_div_rate is a scaling factor which controls the actual division rate
97  if (p.in_tissue && !p.boundary && V.area > m_max_A0)
98  {
99  double prob_div = fact*(V.area - m_max_A0); // Bell model of division
100  if (m_rng->drnd() < prob_div) // Only internal verices can divide
101  {
102  //cout << t << " " << V.area << " " << p.A0 << " " << exp((V.area-p.A0)/m_div_rate) << endl;
103  Particle p_new(m_system->size(), p.get_type(), p.get_radius());
104  p_new.x = p.x + m_alpha*m_split_distance*p.get_radius()*p.nx;
105  p_new.y = p.y + m_alpha*m_split_distance*p.get_radius()*p.ny;
106  p_new.z = p.z + m_alpha*m_split_distance*p.get_radius()*p.nz;
107  p_new.set_parent(p.get_flag());
108  m_system->apply_periodic(p_new.x,p_new.y,p_new.z);
109 
110  p.x -= (1.0-m_alpha)*m_split_distance*p.get_radius()*p.nx;
111  p.y -= (1.0-m_alpha)*m_split_distance*p.get_radius()*p.ny;
112  p.z -= (1.0-m_alpha)*m_split_distance*p.get_radius()*p.nz;
113  p.age = 0.0;
114  p.A0 = p.get_A0();
115  m_system->apply_periodic(p.x,p.y,p.z);
116 
117  p_new.nx = p.nx; p_new.ny = p.ny; p_new.nz = p.nz;
118  p_new.vx = p.vx; p_new.vy = p.vy; p_new.vz = p.vz;
119  p_new.Nx = p.Nx; p_new.Ny = p.Ny; p_new.Nz = p.Nz;
120  p_new.age = 0.0;
121  p_new.A0 = p.get_A0();
122  p_new.set_radius(p.get_radius());
123  p_new.set_type(p.get_type());
124  p_new.set_default_area(p.get_A0());
125  p_new.in_tissue = true;
126  for(list<string>::iterator it_g = p.groups.begin(); it_g != p.groups.end(); it_g++)
127  p_new.add_group(*it_g);
128  m_system->add_particle(p_new);
129  }
130  }
131  }
132  if (!m_system->group_ok(m_group_name))
133  {
134  cout << "After cell division: Group info mismatch for group : " << m_group_name << endl;
135  throw runtime_error("Group mismatch.");
136  }
137  m_system->set_force_nlist_rebuild(true);
138  }

149 void PopulationCell::remove(int t)
150 {
151  if (m_freq > 0 && t % m_freq == 0 && m_death_rate > 0.0) // Attempt removal only at certain time steps
152  {
153  double fact = m_freq*m_system->get_integrator_step();
154  if (!m_system->group_ok(m_group_name))
155  {
156  cout << "Before Cell Remove: Group info mismatch for group : " << m_group_name << endl;
157  throw runtime_error("Group mismatch.");
158  }
159  int N = m_system->get_group(m_group_name)->get_size();
160  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
161  vector<int> to_remove;
162  for (int i = 0; i < N; i++)
163  {
164  int pi = particles[i];
165  Particle& p = m_system->get_particle(pi);
166  // actual probability of dying now: (attempt_freq * dt) exp[(age-max_age)*m_death_rate]
167  // In the limits where we can linearise this it gives:
168  // P_div = (attempt_freq * dt) [1+m_death_rate*(age-age_max)]. This death rate is an inverse time scale
169  //double prob_death =m_freq*m_system->get_integrator_step()*exp((p.age-m_max_age)*m_death_rate);
170  // Trying a very simple, linearly increasing death chance
171  //double prob_death = fact*p.age/m_max_age;
172  double prob_death = m_death_rate*m_freq*m_system->get_integrator_step(); // actual probability of dividing now: rate * (attempt_freq * dt)
173  if (p.in_tissue && !p.boundary && m_rng->drnd() < prob_death)
174  to_remove.push_back(p.get_id());
175  }
176  int offset = 0;
177  for (vector<int>::iterator it = to_remove.begin(); it != to_remove.end(); it++)
178  {
179  m_system->remove_particle((*it)-offset);
180  offset++;
181  }
182  if (m_system->size() == 0)
183  {
184  m_msg->msg(Messenger::ERROR,"Cell population control. No cells left in the system. Please reduce that death rate.");
185  throw runtime_error("No cells left in the system.");
186  }
187  if (!m_system->group_ok(m_group_name))
188  {
189  cout << "After Cell Remove: Group info mismatch for group : " << m_group_name << endl;
190  throw runtime_error("Group mismatch.");
191  }
192  m_system->set_force_nlist_rebuild(true);
193  }
194 }

201 void PopulationCell::grow(int t)
202 {
203  if (m_freq > 0 && t % m_freq == 0 && m_growth_rate > 0.0)
204  {
205  double fact = m_freq*m_system->get_integrator_step()*m_growth_rate;
206  int N = m_system->get_group(m_group_name)->get_size();
207  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
208  for (int i = 0; i < N; i++)
209  {
210  // Growth probability stays dimensionless, between 0 and 1. Instead, the actual growth rate is no an inverse time
211  if (m_rng->drnd() < m_growth_prob)
212  {
213  int pi = particles[i];
214  Particle& p = m_system->get_particle(pi);
215  if (p.in_tissue)
216  p.A0 *= (1.0+fact);
217  }
218  }
219  m_system->set_force_nlist_rebuild(true);
220  if (m_rescale_contacts)
221  m_system->set_nlist_rescale(sqrt(1.0+fact));
222  }
