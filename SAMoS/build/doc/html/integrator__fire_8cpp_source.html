58  p.y += m_dt*p.vy;

64  double dtheta = m_dt*p.omega; //m_constraint->project_torque(p);
65  m_constrainer->rotate_director(p,dtheta);
66  }
67

70  m_system->reset_torques();
71  // compute forces in the current configuration
72  if (m_potential)

74  // compute torques in the current configuration
75  if (m_align)
76  m_align->compute();
77

115  {

118  p.vx = fact_1*p.vx + fact_2 * p.fx;
119  p.vy = fact_1*p.vy + fact_2 * p.fy;
120  p.vz = fact_1*p.vz + fact_2 * p.fz;
121  }
