34 void ConstraintHourglass::compute_normal(Particle& p, double& Nx, double& Ny, double& Nz)
35 {
36  // Compute unit gradient of the implicit function (normal will be normalized gradient)
37  this->compute_gradient(p,Nx,Ny,Nz);
38  // Normalize N
39  double len_N = sqrt(Nx*Nx + Ny*Ny + Nz*Nz);
40  Nx /= len_N; Ny /= len_N; Nz /= len_N;
41  p.Nx = Nx; p.Ny = Ny; p.Nz = Nz;
42 }

47 void ConstraintHourglass::compute_gradient(Particle&p, double& gx, double& gy, double& gz)
48 {
49  double x = p.x, y = p.y, z = p.z;
50  BoxPtr box = m_system->get_box();
51  double lz = box->Lz;
52  gx = 2.0*x;
53  gy = 2.0*y;
54  gz = (-4*m_A*m_n*M_PI*cos((2*m_n*M_PI*z)/lz)*(m_R + m_A*sin((2*m_n*M_PI*z)/lz)))/lz;
55 }

60 double ConstraintHourglass::constraint_value(Particle& p)
61 {
62  double x = p.x, y = p.y, z = p.z;
63  BoxPtr box = m_system->get_box();
64  double lz = box->Lz;
65  double zfact = m_R + m_A*sin(2.0*M_PI/lz*m_n*z);
66  double g = x*x + y*y - zfact*zfact;
67  return g;
