33 void ExternalFieldAlign::compute()
34 {
35  int N = m_system->size();
36  double hx = m_hx, hy = m_hy, hz = m_hz;
37 
38  for (int i = 0; i < N; i++)
39  {
40  Particle& pi = m_system->get_particle(i);
41  if (m_has_params)
42  {
43  hx = m_type_params[pi.get_type()-1].hx;
44  hy = m_type_params[pi.get_type()-1].hy;
45  hz = m_type_params[pi.get_type()-1].hz;
46  }
47  pi.tau_x += hy*pi.nz - hz*pi.ny;
48  pi.tau_y += hz*pi.nx - hx*pi.nz;
49  pi.tau_z += hx*pi.ny - hy*pi.nx;
50  }
51 }
