33 void ExternalBoundaryPull::compute()
34 {
35  int N = m_system->size();
36  double alpha = m_alpha;
37 
38  // First we compute centre of mass to be able to determine outward direction
39  double xcm = 0.0, ycm = 0.0, zcm = 0.0;
40  int cnt = 0;
41  for (int i = 0; i < N; i++)
42  {
43  Particle& pi = m_system->get_particle(i);
44  if (pi.in_tissue)
45  {
46  xcm += pi.x;
47  ycm += pi.y;
48  zcm += pi.z;
49  cnt++;
50  }
51  }
52  xcm /= cnt; ycm /= cnt; zcm /= cnt;
53 
54  for (int i = 0; i < N; i++)
55  {
56  Particle& pi = m_system->get_particle(i);
57  if (pi.boundary)
58  {
59  double force_sign = 1.0;
60  double X = pi.x - xcm, Y = pi.y - ycm, Z = pi.z - zcm;
61  double len_R = sqrt(X*X + Y*Y + Z*Z);
62  X /= len_R; Y /= len_R; Y /= len_R;
63  Particle& pj = m_system->get_particle(pi.boundary_neigh[0]);
64  Particle& pk = m_system->get_particle(pi.boundary_neigh[1]);
65  double xji = pj.x - pi.x, yji = pj.y - pi.y, zji = pj.z - pi.z;
66  double len_ji = sqrt(xji*xji + yji*yji + zji*zji);
67  xji /= len_ji; yji /= len_ji; zji /= len_ji;
68  double xki = pk.x - pi.x, yki = pk.y - pi.y, zki = pk.z - pi.z;
69  double len_ki = sqrt(xki*xki + yki*yki + zki*zki);
70  xki /= len_ki; yki /= len_ki; zki /= len_ki;
71  double x = -(xji+xki), y = -(yji+yki), z = -(zji+zki);
72  double len_r = sqrt(x*x + y*y + z*z);
73  x /= len_r; y /= len_r; z /= len_r;
74  // compute dot product with radius the vector connecting center of mass and pi
75  if ((x*X + y*Y + z*Z) < 0.0)
76  force_sign = -1.0;
77  double factor = force_sign*alpha;
78  pi.fx += factor*x;
79  pi.fy += factor*y;
80  pi.fz += factor*z;
81  }
82  }
83 }
