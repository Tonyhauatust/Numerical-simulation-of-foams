172  double k = 1.0 - c;

174  double nx = x * (c + k * v.x * v.x) + y * (k * v.x * v.y - s * v.z) + z * (k * v.x * v.z + s * v.y);
175  double ny = x * (k * v.x * v.y + s * v.z) + y * (c + k * v.y * v.y) + z * (k * v.y * v.z - s * v.x);
176  double nz = x * (k * v.x * v.z - s * v.y) + y * (k * v.y * v.z + s * v.x) + z * (c + k * v.z * v.z);
177
