33 void PairRodPotential::compute(double dt)
34 {
35  int N = m_system->size();
36  double k = m_k;
37  double push = m_push;
38  double ai, aj;
39  double li2, lj2;
40  double lambda, mu;
41  double dx, dy, dz;
42  double fx, fy, fz;
43  double force_factor;
44  double pot_eng;
45  double alpha_i = 1.0; // phase in factor for particle i
46  double alpha_j = 1.0; // phase in factor for particle j
47  double alpha = 1.0; // phase in factor for pair interaction (see below)
48 
49  if (m_system->compute_per_particle_energy())
50  {
51  for (int i = 0; i < N; i++)
52  {
53  Particle& p = m_system->get_particle(i);
54  p.set_pot_energy("rod",0.0);
55  }
56  }
57 
58  m_potential_energy = 0.0;
59  for (int i = 0; i < N; i++)
60  {
61  Particle& pi = m_system->get_particle(i);
62  if (m_phase_in)
63  alpha_i = 0.5*(1.0 + m_val->get_val(static_cast<int>(pi.age/dt)));
64  ai = pi.get_radius();
65  li2 = 0.5*pi.get_length();
66  double ni_x = pi.nx, ni_y = pi.ny, ni_z = pi.nz;
67  vector<int>& neigh = m_nlist->get_neighbours(i);
68  for (unsigned int j = 0; j < neigh.size(); j++)
69  {
70  Particle& pj = m_system->get_particle(neigh[j]);
71  if (m_phase_in)
72  {
73  alpha_j = 0.5*(1.0 + m_val->get_val(static_cast<int>(pj.age/dt)));
74  // Determine global phase in factor: particles start at 0.5 strength (both daugthers of a division replace the mother)
75  // Except for the interaction between daugthers which starts at 0
76  if (alpha_i<1.0 && alpha_j < 1.0)
77  alpha = alpha_i + alpha_j - 1.0;
78  else
79  alpha = alpha_i*alpha_j;
80  }
81  if (m_has_pair_params)
82  {
83  k = m_pair_params[pi.get_type()-1][pj.get_type()-1].k;
84  push = m_pair_params[pi.get_type()-1][pj.get_type()-1].push;
85  }
86  aj = pj.get_radius();
87  lj2 = 0.5*pj.get_length();
88  double nj_x = pj.nx, nj_y = pj.ny, nj_z = pj.nz;
89  double dx_cm = pj.x - pi.x, dy_cm = pj.y - pi.y, dz_cm = pj.z - pi.z;
90  m_system->apply_periodic(dx_cm,dy_cm,dz_cm);
91 
92  double ni_dot_nj = ni_x*nj_x + ni_y*nj_y + ni_z*nj_z;
93  double drcm_dot_ni = dx_cm*ni_x + dy_cm*ni_y + dz_cm*ni_z;
94  double drcm_dot_nj = dx_cm*nj_x + dy_cm*nj_y + dz_cm*nj_z;
95  double cc = 1.0 - ni_dot_nj*ni_dot_nj;
96 
97 
98  if (cc <= 1e-10) // rods are nearly parallel
99  {
100  if (fabs(drcm_dot_ni) > 1e-10)
101  {
102  lambda = copysign(li2,drcm_dot_ni);
103  mu = lambda*ni_dot_nj - drcm_dot_nj;
104  if (fabs(mu) > lj2) mu = copysign(lj2,mu);
105  }
106  else
107  {
108  lambda = 0.0;
109  mu = 0.0;
110  }
111 // lambda = 0.5*drcm_dot_ni;
112 // mu = -0.5*drcm_dot_nj;
113 // if (fabs(lambda) > li2) lambda = copysign(li2,lambda);
114 // if (fabs(mu) > lj2) mu = copysign(lj2,mu);
115  }
116  else
117  {
118  lambda = (drcm_dot_ni-ni_dot_nj*drcm_dot_nj)/cc;
119  mu = (-drcm_dot_nj+ni_dot_nj*drcm_dot_ni)/cc;
120  if (fabs(lambda) > li2 || fabs(mu) > lj2)
121  {
122  if (fabs(lambda) - li2 > fabs(mu) - lj2)
123  {
124  lambda = copysign(li2,lambda);
125  mu = lambda*ni_dot_nj - drcm_dot_nj;
126  if (fabs(mu) > lj2) mu = copysign(lj2,mu);
127  }
128  else
129  {
130  mu = copysign(lj2,mu);
131  lambda = mu*ni_dot_nj + drcm_dot_ni;
132  if (fabs(lambda) > li2) lambda = copysign(li2,lambda);
133  }
134  }
135  }
136 
137  dx = dx_cm + mu*nj_x - lambda*ni_x;
138  dy = dy_cm + mu*nj_y - lambda*ni_y;
139  dz = dz_cm + mu*nj_z - lambda*ni_z;
140 
141  double r_sq = dx*dx + dy*dy + dz*dz;
142  double r = sqrt(r_sq);
143  double ai_p_aj = ai+aj;
144 
145  if (r < ai_p_aj)
146  {
147  // Handle potential
148  double diff = ai_p_aj - r;
149  pot_eng = 0.5*k*alpha*diff*diff;
150  if (m_model == "hertz")
151  pot_eng *= 0.8*sqrt(diff);
152  m_potential_energy += pot_eng;
153  // Handle force
154 
155  if (r >= SMALL_NUMBER)
156  {
157  force_factor = k*alpha*diff/r;
158  if (m_model == "hertz")
159  force_factor *= sqrt(diff);
160  fx = force_factor*dx; fy = force_factor*dy; fz = force_factor*dz;
161  }
162  else
163  {
164  r_sq = dx_cm*dx_cm + dy_cm*dy_cm + dz_cm*dz_cm;
165  r = sqrt(r_sq);
166  if (r > 0.0)
167  force_factor = k*alpha*push*diff/r;
168  else
169  force_factor = 0.0;
170  if (m_model == "hertz")
171  force_factor *= sqrt(diff);
172  fx = force_factor*dx_cm; fy = force_factor*dy_cm ; fz = force_factor*dz_cm;
173  }
174  pi.fx -= fx;
175  pi.fy -= fy;
176  pi.fz -= fz;
177  // Use 3d Newton's law
178  pj.fx += fx;
179  pj.fy += fy;
180  pj.fz += fz;
181  // handle torques
182  pi.tau_x -= lambda*(ni_y*fz - ni_z*fy);
183  pi.tau_y -= lambda*(ni_z*fx - ni_x*fz);
184  pi.tau_z -= lambda*(ni_x*fy - ni_y*fx);
185  pj.tau_x += mu*(nj_y*fz - nj_z*fy);
186  pj.tau_y += mu*(nj_z*fx - nj_x*fz);
187  pj.tau_z += mu*(nj_x*fy - nj_y*fx);
188  if (m_system->compute_per_particle_energy())
189  {
190  pi.add_pot_energy("rod",pot_eng);
191  pj.add_pot_energy("rod",pot_eng);
192  }
193  }
194  }
195  }
