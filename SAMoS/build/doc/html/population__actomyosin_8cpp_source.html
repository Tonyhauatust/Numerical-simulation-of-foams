42 void PopulationActomyosin::divide(int t)
43 {
44  if ((m_freq > 0) && (t % m_freq == 0) && (m_attach_rate > 0.0 || m_detach_rate > 0.0)) // Attempt D to A transition only at certain time steps
45  {
46  if (!m_system->group_ok(m_group_name))
47  {
48  cout << "Before actomyosin division: Group info mismatch for group : " << m_group_name << endl;
49  throw runtime_error("Group mismatch.");
50  }
51  int N = m_system->size();
52  vector<int> to_attach;
53  vector<int> to_detach;
54  // Probability of attachment/detachment for a given particle is rate per particle multiplied with time,
55  // where time is equal to m_freq*integrator_time_step.
56  double attach_prob = m_attach_rate*m_freq*m_system->get_integrator_step();
57  double detach_prob = m_detach_rate*m_freq*m_system->get_integrator_step();
58  for (int i = 0; i < N; i++)
59  {
60  Particle& pi = m_system->get_particle(i);
61  if (pi.get_type() == m_type_actin)
62  {
63  vector<int>& neigh = m_nlist->get_neighbours(i);
64  double min_dist = 1e10;
65  int min_id = 0;
66  for (unsigned int j = 0; j < neigh.size(); j++) // Loop over all neigbours of actin bead i
67  {
68  Particle& pj = m_system->get_particle(neigh[j]);
69  if (pj.get_type() == m_type_d) // if the neighbour if of type "detached"
70  {
71  double dx = pi.x - pj.x, dy = pi.y - pj.y, dz = pi.z - pj.z;
72  m_system->apply_periodic(dx,dy,dz);
73  double r = sqrt(dx*dx + dy*dy + dz*dz); // compute distance to the neighbour
74  if (r < min_dist) // if this distance is below current minimum distance record it and the id of the bead itself
75  {
76  min_dist = r;
77  min_id = pj.get_id();
78  }
79  }
80  }
81  if (min_dist < m_re) // If the closest "detached" bead is within cutoff distance re
82  to_attach.push_back(min_id);
83  //if (find(to_attach.begin(),to_attach.end(),min_id) == to_attach.end())
84  // to_attach.push_back(min_id);
85  }
86  if (pi.get_type() == m_type_a)
87  to_detach.push_back(pi.get_id());
88  }
89  for (vector<int>::iterator it_a = to_attach.begin(); it_a != to_attach.end(); it_a++)
90  if (m_rng->drnd() < attach_prob) // flip its type to "attached" with probability attach_prob.
91  {
92  Particle& pi = m_system->get_particle(*it_a);
93  pi.set_type(m_type_a);
94  }
95  for (vector<int>::iterator it_d = to_detach.begin(); it_d != to_detach.end(); it_d++)
96  {
97  Particle& pi = m_system->get_particle(*it_d);
98  // TODO: Figure out how to incorporate forces
99  //double f = sqrt(pi.fx*pi.fx + pi.fy*pi.fy + pi.fz*pi.fz);
100  //double prob = m_detach_prob*exp(m_lambda*f);
101  double f = sqrt(pi.fx*pi.fx + pi.fy*pi.fy + pi.fz*pi.fz);
102  double detach_prob_fdep = detach_prob*exp(m_lambda*f);
103  if (m_rng->drnd() < detach_prob_fdep) // flip its type to "attached" with probability attach_prob.
104  pi.set_type(m_type_d);
105  }
106  }
107 }
