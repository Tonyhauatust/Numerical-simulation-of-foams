35 void ExternalShapeAlign::compute()
36 {
37  int N = m_system->size();
38  double J = m_J;
39  Mesh& mesh = m_system->get_mesh();
40 
41  if (mesh.size() > 0)
42  {
43  vector<Vertex>& vertices = mesh.get_vertices();
44  vector<Face>& faces = mesh.get_faces();
45  double ax, ay; // components of the shape eigenvector
46  for (int i = 0; i < N; i++)
47  {
48  Particle& pi = m_system->get_particle(i);
49  if (!pi.boundary)
50  {
51  Vertex& V = vertices[pi.get_id()];
52  // Compute components of the shape tensor defined by outter product of cell edges
53  double A = 0.0; // xx component
54  double B = 0.0; // xy component
55  double C = 0.0; // yy component
56  for (unsigned int d = 0; d < V.dual.size(); d++)
57  {
58  Face& f1 = faces[V.dual[d]];
59  int d_p_1 = (d == V.dual.size()-1) ? 0 : d + 1;
60  Face& f2 = faces[V.dual[d_p_1]];
61  Vector3d dr = f2.rc - f1.rc;
62  A += dr.x*dr.x; B += dr.x*dr.y; C += dr.y*dr.y;
63  }
64  A /= V.dual.size(); B /= V.dual.size(); C /= V.dual.size();
65  // Maximum eigenvalue is
66  double l1 = 0.5*(A+C + sqrt((A-C)*(A-C)+4*B*B));
67  double l2 = 0.5*(A+C - sqrt((A-C)*(A-C)+4*B*B));
68  double lambda = (l1 > l2) ? l1 : l2;
69  // compute corresponding eigen vector
70  ax = B;
71  ay = lambda-A;
72  // normalise it
73  double len_a = sqrt(ax*ax + ay*ay);
74  ax /= len_a; ay /= len_a;
75  double tau_z = pi.nx*ay - pi.ny*ax;
76  if (m_has_params)
77  J = m_type_params[pi.get_type()-1].J;
78  pi.tau_x += 0.0;
79  pi.tau_y += 0.0;
80  pi.tau_z += J*tau_z; // per construction torque is only in the z direction
81  }
82  }
83  }
84 }
