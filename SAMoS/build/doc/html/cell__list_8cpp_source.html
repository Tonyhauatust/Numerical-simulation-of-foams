37 CellList::CellList(SystemPtr sys, MessengerPtr msg, double cutoff) : m_system(sys), m_msg(msg)
38 {
39  m_nx = static_cast<int>(sys->get_box()->Lx/cutoff);
40  m_ny = static_cast<int>(sys->get_box()->Ly/cutoff);
41  m_nz = static_cast<int>(sys->get_box()->Lz/cutoff);
42  m_size = m_nx*m_ny*m_nz;
43  m_wx = sys->get_box()->Lx/m_nx;
44  m_wy = sys->get_box()->Ly/m_ny;
45  m_wz = sys->get_box()->Lz/m_nz;
46  for (int i = 0; i < m_nx; i++)
47  for (int j = 0; j < m_ny; j++)
48  for (int k = 0; k < m_nz; k++)
49  {
50  int idx = m_ny*m_nz*i + m_nz*j + k;
51  m_cells.push_back(Cell(idx));
52  for (int ix = -1; ix <= 1; ix++)
53  for (int iy = -1; iy <= 1; iy++)
54  for (int iz = -1; iz <= 1; iz++)
55  {
56  int iix = i + ix, iiy = j + iy, iiz = k + iz;
57  if (iix < 0) iix = m_nx - 1;
58  else if (iix == m_nx) iix = 0;
59  if (iiy < 0) iiy = m_ny - 1;
60  else if (iiy == m_ny) iiy = 0;
61  if (iiz < 0) iiz = m_nz - 1;
62  else if (iiz == m_nz) iiz = 0;
63  m_cells[idx].add_neighbour(m_ny*m_nz*iix + m_nz*iiy + iiz);
64  }
65  }
66  m_msg->msg(Messenger::INFO,"Created cell list with "+lexical_cast<string>(m_size)+" cells.");
67  m_msg->msg(Messenger::INFO,"Each cell has dimensions ("+lexical_cast<string>(m_wx)+","+lexical_cast<string>(m_wy)+","+lexical_cast<string>(m_wz)+").");
68 }

82 void CellList::populate()

86  bool periodic = m_system->get_periodic();

100  }
