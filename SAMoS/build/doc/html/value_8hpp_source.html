146  ValueLinear(MessengerPtr msg, pairs_type& param) : Value(msg, param)
147  {
148  m_msg->msg(Messenger::INFO,"Using linear interpolation value object.");
149  if (m_max_val < m_min_val)
150  {
151  m_msg->msg(Messenger::ERROR,"Value object. Minimum value has to be smaller than the maximum value.");
152  throw runtime_error("Value parameters are not properly set.");
153  }
154  if (m_steps <= 0)
155  {
156  m_msg->msg(Messenger::ERROR,"Value object. Number of steps has to be positive non-zero integer.");
157  throw runtime_error("Value parameters. Zero steps given.");
158  }
159  m_dv = (m_max_val - m_min_val)/static_cast<double>(m_steps);
160  m_msg->write_config("value.linear.min_val",lexical_cast<string>(m_min_val));
161  m_msg->write_config("value.linear.max_val",lexical_cast<string>(m_max_val));
162  m_msg->write_config("value.linear.steps",lexical_cast<string>(m_steps));
163  m_msg->write_config("value.linear.dv",lexical_cast<string>(m_dv));
164  }
165
