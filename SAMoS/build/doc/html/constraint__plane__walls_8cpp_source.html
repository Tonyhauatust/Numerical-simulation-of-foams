38 void ConstraintPlaneWalls::enforce(Particle& p)
39 {
40  bool apply = false;
41  if (m_group == "all")
42  apply = true;
43  else
44  apply = (find(p.groups.begin(),p.groups.end(),m_group) != p.groups.end());
45  if (apply)
46  {
47  bool periodic = m_system->get_periodic();
48  double ylo = m_system->get_box()->ylo, yhi = m_system->get_box()->yhi;
49  double ly = yhi - ylo;
50  p.z = 0.0;
51  p.vz = 0.0;
52  p.fz = 0.0;
53  // Check periodic boundary conditions
54  if (periodic)
55  {
56  if (p.y <= ylo) p.y += ly;
57  else if (p.y >= yhi) p.y -= ly;
58  }
59  if (p.x >= m_l)
60  {
61  p.x = m_l;
62  p.vx = -p.vx;
63  }
64  else if (p.x <= -m_l)
65  {
66  p.x = -m_l;
67  p.vx = -p.vx;
68  }
69  }
70 }

78 void ConstraintPlaneWalls::rotate_director(Particle& p, double phi)
79 {
80  bool apply = false;
81  if (m_group == "all")
82  apply = true;
83  else
84  apply = (find(p.groups.begin(),p.groups.end(),m_group) != p.groups.end());
85  if (apply)
86  {
87  // Sine and cosine of the rotation angle
88  double c = cos(phi), s = sin(phi);
89  // Rotation matrix around z axis
90  double Rxx = c, Rxy = -s;
91  double Ryx = s, Ryy = c;
92  // Apply rotation matrix
93  double nx = Rxx*p.nx + Rxy*p.ny;
94  double ny = Ryx*p.nx + Ryy*p.ny;
95  double len = sqrt(nx*nx + ny*ny);
96  // Update particle director
97  p.nx = nx/len;
98  p.ny = ny/len;
99  }
100 }
