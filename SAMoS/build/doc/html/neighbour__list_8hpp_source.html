124  NeighbourList(SystemPtr sys, MessengerPtr msg, double cutoff, double pad, pairs_type& param) : m_system(sys),
125  m_msg(msg),

127  m_pad(pad),
128  m_triangulation(false),

131  m_disable_nlist(false),
132  m_remove_detached(true),
133  m_static_boundary(false)
134  {
135  m_msg->write_config("nlist.cut",lexical_cast<string>(m_cut));
136  m_msg->write_config("nlist.pad",lexical_cast<string>(m_pad));
137  // Check if box is large enough for cell list
138  if (m_system->get_box()->Lx > 2.0*(cutoff+pad) && m_system->get_box()->Ly > 2.0*(cutoff+pad) && m_system->get_box()->Lz > 2.0*(cutoff+pad))
139  {
140  m_use_cell_list = true;
141  m_cell_list = shared_ptr<CellList>(new CellList(m_system,m_msg,cutoff+pad));
142  m_msg->msg(Messenger::INFO,"Using cell lists for neighbour list builds.");
143  m_msg->write_config("nlist.build_type","cell");
144  }
145  else
146  {
147  m_use_cell_list = false;
148  m_msg->msg(Messenger::INFO,"Box dimensions are too small to be able to use cell lists. Neighbour list will be built using N^2 algorithm.");
149  m_msg->write_config("nlist.build_type","n_square");
150  }
151  if (param.find("triangulation") != param.end())
152  {
153  if (m_system->get_periodic())
154  {
155  m_msg->msg(Messenger::ERROR,"Delaunay triangulation is not supported for periodic systems.");
156  throw runtime_error("Delaunay triangulation not supported in periodic systems.");
157  }
158  m_triangulation = true;
159  m_msg->msg(Messenger::INFO,"Faces will be build using Delaunay triangulation.");
160  m_msg->write_config("nlist.triangulation","true");
161  }
162  if (param.find("max_perimeter") == param.end())
163  {
164  m_msg->msg(Messenger::WARNING,"Neighbour list. No maximum face perimeter set. Assuming default value of 20.");
165  m_msg->write_config("nlist.max_perimeter","20.0");
166  m_max_perim = 20.0;
167  }
168  else
169  {
170  m_msg->msg(Messenger::INFO,"Neighbour list. Setting maximum face perimeter to "+param["max_perimeter"]+".");
171  m_msg->write_config("nlist.max_perimeter",param["max_perimeter"]);
172  m_max_perim = lexical_cast<double>(param["max_perimeter"]);
173  }
174  if (param.find("circumcenter") != param.end())
175  {
176  m_msg->msg(Messenger::WARNING,"Neighbour list. Using circumcenters for mesh dual.");
177  m_msg->write_config("nlist.circumcenter","true");
178  m_circumcenter = true;
179  }
180  else
181  {
182  m_msg->msg(Messenger::INFO,"Neighbour list. Using geometric centers for mesh dual.");
183  m_msg->write_config("nlist.circumcenter","false");
184  m_circumcenter = false;
185  }
186  if (param.find("disable_nlist") != param.end())
187  {
188  m_msg->msg(Messenger::WARNING,"Neighbour list will not be built. This should be used in pair with triangulations when particle connections are build based on tringulations.");
189  m_msg->write_config("nlist.disable_nlist","true");
190  m_disable_nlist = true;
191  }
192  if (param.find("keep_detached") != param.end())
193  {
194  m_msg->msg(Messenger::WARNING,"Neighbour list. Particles detached from the mesh (tissue) will be kept.");
195  m_msg->write_config("nlist.remove_detached","false");
196  m_remove_detached = false;
197  }
198  if (param.find("max_iter") != param.end())
199  {
200  m_msg->msg(Messenger::INFO,"Neighbour list. Setting maximum number of iterations for boundary builds in tissue simulations to "+param["max_iter"]+".");
201  m_msg->write_config("nlist.max_iter",param["max_iter"]);
202  m_system->set_max_mesh_iterations(lexical_cast<int>(param["max_iter"]));
203  }
204  if (param.find("boundary_type") != param.end())
205  {
206  m_msg->msg(Messenger::INFO,"Neighbour list. Setting type of boundary particles in tissue simulations to "+param["boundary_type"]+".");
207  m_msg->write_config("nlist.boundary_type",param["boundary_type"]);
208  m_system->set_boundary_type(lexical_cast<int>(param["boundary_type"]));
209  }
210  if (param.find("static_boundary") != param.end())
211  {
212  m_msg->msg(Messenger::INFO,"Neighbour list. Using static boundaries in the tissue simulation.");
213  m_msg->write_config("nlist.static_boundary","true");
214  m_static_boundary = true;
215  }
216  this->build();
217  }
