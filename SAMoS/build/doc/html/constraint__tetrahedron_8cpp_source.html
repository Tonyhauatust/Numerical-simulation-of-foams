46 void ConstraintTetrahedron::compute_gradient(Particle&p, double& gx, double& gy, double& gz)
47 {
48  Vector3d pos = Vector3d(p.x,p.y,p.z);
49  Vector3d p_a1 = pos - m_a1;
50  Vector3d p_a2 = pos - m_a2;
51  Vector3d p_a3 = pos - m_a3;
52  Vector3d p_a4 = pos - m_a4;
53  double l1 = p_a1.len();
54  double l2 = p_a2.len();
55  double l3 = p_a3.len();
56  double l4 = p_a4.len();
57  double l1_3 = l1*l1*l1, l2_3 = l2*l2*l2, l3_3 = l3*l3*l3, l4_3 = l4*l4*l4;
58  gx = -(p_a1.x/l1_3 + p_a2.x/l2_3 + p_a3.x/l3_3 + p_a4.x/l4_3);
59  gy = -(p_a1.y/l1_3 + p_a2.y/l2_3 + p_a3.y/l3_3 + p_a4.y/l4_3);
60  gz = -(p_a1.z/l1_3 + p_a2.z/l2_3 + p_a3.z/l3_3 + p_a4.z/l4_3);
61 }

70 double ConstraintTetrahedron::constraint_value(Particle& p)
71 {
72  Vector3d pos = Vector3d(p.x,p.y,p.z);
73  Vector3d p_a1 = pos - m_a1;
74  Vector3d p_a2 = pos - m_a2;
75  Vector3d p_a3 = pos - m_a3;
76  Vector3d p_a4 = pos - m_a4;
77  double g = 1.0/sqrt(dot(p_a1, p_a1));
78  g += 1.0/sqrt(dot(p_a2, p_a2));
79  g += 1.0/sqrt(dot(p_a3, p_a3));
80  g += 1.0/sqrt(dot(p_a4, p_a4));
81  return g - m_value;
82 }

88 bool ConstraintTetrahedron::rescale()
89 {
90  if (m_rescale != 1.0)
91  {
92  int step = m_system->get_step();
93  if ((step % m_rescale_freq == 0) && (step < m_rescale_steps))
94  {
95  m_s *= m_scale;
96  m_a1 = m_s*tetra_a1;
97  m_a2 = m_s*tetra_a2;
98  m_a3 = m_s*tetra_a3;
99  m_a4 = m_s*tetra_a4;
100  for (int i = 0; i < m_system->size(); i++)
101  {
102  Particle& p = m_system->get_particle(i);
103  this->enforce(p);
104  }
105  return true;
106  }
107  }
108  return false;
109 }
