32 void PairCoulombPotential::compute(double dt)
33 {
34  int N = m_system->size();
35  double sigma = m_sigma;
36  double alpha = m_alpha;
37  double sigma_sq = sigma*sigma;
38  double phase_fact_i = 1.0; // phase in factor for particle i
39  double phase_fact_j = 1.0; // phase in factor for particle j
40  double phase_fact = 1.0; // phase in factor for pair interaction (see below)
41 
42  if (m_system->compute_per_particle_energy())
43  {
44  for (int i = 0; i < N; i++)
45  {
46  Particle& p = m_system->get_particle(i);
47  p.set_pot_energy("coulomb",0.0);
48  }
49  }
50 
51  m_potential_energy = 0.0;
52  for (int i = 0; i < N; i++)
53  {
54  Particle& pi = m_system->get_particle(i);
55  if (m_phase_in)
56  phase_fact_i = 0.5*(1.0 + m_val->get_val(static_cast<int>(pi.age/dt)));
57  for (int j = i+1; j < N; j++)
58  {
59  Particle& pj = m_system->get_particle(j);
60  if (m_phase_in)
61  {
62  phase_fact_j = 0.5*(1.0 + m_val->get_val(static_cast<int>(pj.age/dt)));
63  // Determine global phase in factor: particles start at 0.5 strength (both daughters of a division replace the mother)
64  // Except for the interaction between daughters which starts at 0
65  if ( phase_fact_i < 1.0 && phase_fact_j < 1.0)
66  phase_fact=phase_fact_i + phase_fact_j - 1.0;
67  else
68  phase_fact = phase_fact_i*phase_fact_j;
69  }
70  if (m_has_pair_params)
71  {
72  int pi_t = pi.get_type() - 1, pj_t = pj.get_type() - 1;
73  alpha = m_pair_params[pi_t][pj_t].alpha;
74  sigma = m_pair_params[pi_t][pj_t].sigma;
75  sigma_sq = sigma*sigma;
76  }
77  double dx = pj.x - pi.x, dy = pj.y - pi.y, dz = pj.z - pi.z;
78  m_system->apply_periodic(dx,dy,dz);
79  double r_sq = dx*dx + dy*dy + dz*dz;
80  double r = sqrt(r_sq);
81  double inv_r_sq = sigma_sq/r_sq;
82  double inv_r_6 = inv_r_sq*inv_r_sq*inv_r_sq;
83  // Handle potential
84  double potential_energy = phase_fact*(alpha/r + 4.0*fabs(alpha)*inv_r_6*inv_r_6);
85  m_potential_energy += potential_energy;
86  // Handle force
87  double r_3 = r*r_sq;
88  double force_factor = phase_fact*(alpha/r_3 + 48.0*fabs(alpha)*inv_r_6*inv_r_6*inv_r_sq);
89  pi.fx -= force_factor*dx;
90  pi.fy -= force_factor*dy;
91  pi.fz -= force_factor*dz;
92  // Use 3d Newton's law
93  pj.fx += force_factor*dx;
94  pj.fy += force_factor*dy;
95  pj.fz += force_factor*dz;
96  if (m_system->compute_per_particle_energy())
97  {
98  pi.add_pot_energy("coulomb",potential_energy);
99  pj.add_pot_energy("coulomb",potential_energy);
100  }
101  }
102  }
103 }
