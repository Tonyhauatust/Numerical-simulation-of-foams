37 void IntegratorSepulveda::integrate()
38 {
39  int N = m_system->get_group(m_group_name)->get_size();
40  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
41  double dt_2 = 0.5*m_dt;
42  double B = sqrt(m_tau*m_dt);
43  double theta = 1.0 - m_dt;
44 
45  // Generalized Langevin Dynamics, pg. 377 step 1
46  // Perform first half step for velocity
47  for (int i = 0; i < N; i++)
48  {
49  int pi = particles[i];
50  Particle& p = m_system->get_particle(pi);
51  //Particle& p = m_system->get_particle(i);
52  p.vx += dt_2*(p.fx + m_eta_x[i])/p.mass;
53  p.vy += dt_2*(p.fy + m_eta_y[i])/p.mass;
54  p.vz += dt_2*(p.fz + m_eta_z[i])/p.mass;
55  // Project everything back to the manifold
56  m_constrainer->enforce(p);
57  // Update angular velocity
58  p.omega += dt_2*m_constrainer->project_torque(p);
59  }
60  // Generalized Langevin Dynamics, pg. 377 step 2
61  // update position
62  for (int i = 0; i < N; i++)
63  {
64  int pi = particles[i];
65  Particle& p = m_system->get_particle(pi);
66  //Particle& p = m_system->get_particle(i);
67  p.x += m_dt*p.vx;
68  p.y += m_dt*p.vy;
69  p.z += m_dt*p.vz;
70  // Project everything back to the manifold
71  m_constrainer->enforce(p);
72  }
73  // Generalized Langevin Dynamics, pg. 377 step 3
74  // Orstein-Uhlenbeck for etas
75  // Note: in the book what we call m_alpha is called c_k
76  // and what we call B is called alpha sqrt(2KbT c_k)
77  for (int i = 0; i < N; i++)
78  {
79  // This computes realtive velocity with all neighbours
80  // ------------------------------
81  int pi = particles[i];
82  Particle& p = m_system->get_particle(pi);
83  double dvx = 0.0, dvy = 0.0, dvz = 0.0;
84  vector<int>& neigh = m_nlist->get_neighbours(i);
85  for (unsigned int j = 0; j < neigh.size(); j++)
86  {
87  Particle& pj = m_system->get_particle(neigh[j]);
88  dvx += pj.vx - p.vx;
89  dvy += pj.vy - p.vy;
90  dvz += pj.vz - p.vz;
91  }
92  // ------------------------------
93  m_eta_x[i] = theta*m_eta_x[i] + (1-theta)*(-m_alpha*p.vx + dvx) + B*m_rng->gauss_rng(1.0);
94  m_eta_y[i] = theta*m_eta_y[i] + (1-theta)*(-m_alpha*p.vy + dvy) + B*m_rng->gauss_rng(1.0);
95  m_eta_z[i] = theta*m_eta_z[i] + (1-theta)*(-m_alpha*p.vz + dvz) + B*m_rng->gauss_rng(1.0);
96  }
97 
98  // Enforce constraints and update alignment
99  for (int i = 0; i < N; i++)
100  {
101  int pi = particles[i];
102  Particle& p = m_system->get_particle(pi);
103  //Particle& p = m_system->get_particle(i);
104  // Change orientation of the velocity (in the tangent plane) according to eq. (1b)
105  double dtheta = m_dt*p.omega; //m_constraint->project_torque(p);
106  m_constrainer->rotate_director(p,dtheta);
107  //p.omega = dtheta*m_dt;
108  }
109 
110  // reset forces and torques
111  m_system->reset_forces();
112  m_system->reset_torques();
113 
114 
115  // compute forces in the current configuration
116  if (m_potential)
117  m_potential->compute(m_dt);
118  // compute torques in the current configuration
119  if (m_align)
120  m_align->compute();
121 
122  // Generalized Langevin Dynamics, pg. 377 step 4
123  // Perform second half step for velocity only if there is no limit on particle move
124  for (int i = 0; i < N; i++)
125  {
126  int pi = particles[i];
127  Particle& p = m_system->get_particle(pi);
128  //Particle& p = m_system->get_particle(i);
129  p.vx += dt_2*(p.fx + m_eta_x[i])/p.mass;
130  p.vy += dt_2*(p.fy + m_eta_y[i])/p.mass;
131  p.vz += dt_2*(p.fz + m_eta_z[i])/p.mass;
132 
133  // Project everything back to the manifold
134  m_constrainer->enforce(p);
135  // Update angular velocity
136  p.omega += dt_2*m_constrainer->project_torque(p);
137  p.age += m_dt;
138  }
139  // Update vertex mesh
140  m_system->update_mesh();
141 }
