32 void PairVicsekAlign::compute()
33 {
34  int N = m_system->size();
35  bool periodic = m_system->get_periodic();
36  BoxPtr box = m_system->get_box();
37  double rcut = m_rcut;
38 
39  for (int i = 0; i < N; i++)
40  {
41  Particle& pi = m_system->get_particle(i);
42  vector<int>& neigh = m_nlist->get_neighbours(i);
43  for (unsigned int j = 0; j < neigh.size(); j++)
44  {
45  Particle& pj = m_system->get_particle(neigh[j]);
46  double dx = pj.x - pi.x, dy = pj.y - pi.y, dz = pj.z - pi.z;
47  if (periodic)
48  {
49  if (dx > box->xhi) dx -= box->Lx;
50  else if (dx < box->xlo) dx += box->Lx;
51  if (dy > box->yhi) dy -= box->Ly;
52  else if (dy < box->ylo) dy += box->Ly;
53  if (dz > box->zhi) dz -= box->Lz;
54  else if (dz < box->zlo) dz += box->Lz;
55  }
56  double r_sq = dx*dx + dy*dy + dz*dz;
57  if (r_sq <= rcut*rcut)
58  {
59  pi.tau_x += pj.vx;
60  pi.tau_y += pj.vy;
61  pi.tau_z += pj.vz;
62  // Since we use 3d Newton's and only have half the the neighbour list we need handle the other part as well
63  pj.tau_x += pi.vx;
64  pj.tau_y += pi.vy;
65  pj.tau_z += pi.vz;
66  }
67  }
68  pi.tau_x += pi.vx;
69  pi.tau_y += pi.vy;
70  pi.tau_z += pi.vz;
71  }
72  // Now we need to normalize all taus
73  for (int i = 0; i < N; i++)
74  {
75  Particle& p = m_system->get_particle(i);
76  double v = sqrt(p.tau_x*p.tau_x + p.tau_y*p.tau_y + p.tau_z*p.tau_z);
77  if (v == 0.0) v = 1.0;
78  double inv_v = 1.0/v;
79  p.tau_x *= inv_v;
80  p.tau_y *= inv_v;
81  p.tau_z *= inv_v;
82  }
83 }
