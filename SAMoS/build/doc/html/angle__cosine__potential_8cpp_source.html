33 {

35  bool periodic = m_system->get_periodic();
36  BoxPtr box = m_system->get_box();
37  double k = m_k;
38 
39  m_potential_energy = 0.0;
40  for (int i = 0; i < Nangles; i++)
41  {
42  Angle& a = m_system->get_angle(i);
43  Particle& pi = m_system->get_particle(a.i);
44  Particle& pj = m_system->get_particle(a.j);
45  Particle& pk = m_system->get_particle(a.k);
46  double dx1 = pi.x - pj.x, dy1 = pi.y - pj.y, dz1 = pi.z - pj.z;
47  double dx2 = pk.x - pj.x, dy2 = pk.y - pj.y, dz2 = pk.z - pj.z;
48  if (periodic)
49  {
50  if (dx1 > box->xhi) dx1 -= box->Lx;
51  else if (dx1 < box->xlo) dx1 += box->Lx;
52  if (dy1 > box->yhi) dy1 -= box->Ly;
53  else if (dy1 < box->ylo) dy1 += box->Ly;
54  if (dz1 > box->zhi) dz1 -= box->Lz;
55  else if (dz1 < box->zlo) dz1 += box->Lz;
56  if (dx2 > box->xhi) dx2 -= box->Lx;
57  else if (dx2 < box->xlo) dx2 += box->Lx;
58  if (dy2 > box->yhi) dy2 -= box->Ly;
59  else if (dy2 < box->ylo) dy2 += box->Ly;
60  if (dz2 > box->zhi) dz2 -= box->Lz;
61  else if (dz2 < box->zlo) dz2 += box->Lz;
62  }
63  if (m_has_angle_params)
64  {
65  k = m_angle_params[a.type-1].k;
66  }
67 
68  double r_sq_1 = dx1*dx1 + dy1*dy1 + dz1*dz1;
69  double r_sq_2 = dx2*dx2 + dy2*dy2 + dz2*dz2;
70  double r_1 = sqrt(r_sq_1);
71  double r_2 = sqrt(r_sq_2);

73  c /= r_1*r_2;
74
