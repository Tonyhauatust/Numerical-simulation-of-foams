32 void PairMorsePotential::compute(double dt)
33 {
34  int N = m_system->size();
35  double D = m_D;
36  double a = m_a;
37  double re = m_re;
38  double rcut = m_rcut;
39  double rcut_sq = rcut*rcut;
40  double alpha_i = 1.0; // phase in factor for particle i
41  double alpha_j = 1.0; // phase in factor for particle j
42  double alpha = 1.0; // phase in factor for pair interaction (see below)
43 
44  if (m_system->compute_per_particle_energy())
45  {
46  for (int i = 0; i < N; i++)
47  {
48  Particle& p = m_system->get_particle(i);
49  p.set_pot_energy("morse",0.0);
50  }
51  }
52 
53  // Reset total potential energy to zero
54  m_potential_energy = 0.0;
55  for (int i = 0; i < N; i++)
56  {
57  Particle& pi = m_system->get_particle(i);
58  if (m_phase_in)
59  alpha_i = 0.5*(1.0 + m_val->get_val(static_cast<int>(pi.age/dt)));
60  double ai = pi.get_radius();
61  vector<int>& neigh = m_nlist->get_neighbours(i);
62  for (unsigned int j = 0; j < neigh.size(); j++)
63  {
64  Particle& pj = m_system->get_particle(neigh[j]);
65  if (m_phase_in)
66  {
67  alpha_j = 0.5*(1.0 + m_val->get_val(static_cast<int>(pj.age/dt)));
68  // Determine global phase in factor: particles start at 0.5 strength (both daugthers of a division replace the mother)
69  // Except for the interaction between daugthers which starts at 0
70  if (alpha_i < 1.0 && alpha_j < 1.0)
71  alpha = alpha_i + alpha_j - 1.0;
72  else
73  alpha = alpha_i*alpha_j;
74  }
75  if (m_has_pair_params)
76  {
77  rcut = m_pair_params[pi.get_type()-1][pj.get_type()-1].rcut;
78  rcut_sq = rcut*rcut;
79  }
80  double dx = pi.x - pj.x, dy = pi.y - pj.y, dz = pi.z - pj.z;
81  m_system->apply_periodic(dx,dy,dz);
82  double r_sq = dx*dx + dy*dy + dz*dz;
83  if (r_sq <= rcut_sq)
84  {
85  if (m_has_pair_params)
86  {
87  int pi_t = pi.get_type() - 1, pj_t = pj.get_type() - 1;
88  D = m_pair_params[pi_t][pj_t].D;
89  a = m_pair_params[pi_t][pj_t].a;
90  re = m_pair_params[pi_t][pj_t].re;
91  }
92  if (m_use_particle_radii)
93  re = ai + pj.get_radius();
94  double r = sqrt(r_sq);
95  double exp_fact = exp(-a*(r-re));
96  double pot_fact = exp_fact - 1.0;
97  // Handle potential
98  double potential_energy = D*alpha*(pot_fact*pot_fact-1.0);
99  if (m_shifted)
100  {
101  double shift_fact = exp(-a*(rcut-re)) - 1.0;
102  potential_energy -= D*alpha*(shift_fact*shift_fact-1.0);
103  }
104  m_potential_energy += potential_energy;
105  // Handle force
106  double force_factor = 2.0*D*a*alpha*exp_fact*pot_fact/r;
107  pi.fx += force_factor*dx;
108  pi.fy += force_factor*dy;
109  pi.fz += force_factor*dz;
110  // Use 3d Newton's law
111  pj.fx -= force_factor*dx;
112  pj.fy -= force_factor*dy;
113  pj.fz -= force_factor*dz;
114  if (m_system->compute_per_particle_energy())
115  {
116  pi.add_pot_energy("morse",potential_energy);
117  pj.add_pot_energy("morse",potential_energy);
118  }
119  }
120  }
121  }
122 }
