76  Vector3d P_m_Q = P - Q;
77  double qx = Q.x + 2.0*((1.0 - n_xx)*P_m_Q.x - n_xy*P_m_Q.y - n_xz*P_m_Q.z);
78  double qy = Q.y + 2.0*( - n_yx*P_m_Q.x + (1.0 - n_yy)*P_m_Q.y - n_yz*P_m_Q.z);
79  double qz = Q.z + 2.0*( - n_zx*P_m_Q.x - n_zy*P_m_Q.y + (1.0 - n_zz)*P_m_Q.z);
