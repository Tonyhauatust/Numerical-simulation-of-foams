38 void ConstraintCylinder::enforce(Particle& p)
39 {
40  bool apply = false;
41  if (m_group == "all")
42  apply = true;
43  else
44  apply = (find(p.groups.begin(),p.groups.end(),m_group) != p.groups.end());
45  if (apply)
46  {
47  bool periodic = m_system->get_periodic();
48  BoxPtr box = m_system->get_box();
49  double x = p.x, y = p.y;
50  double R = sqrt(x*x + y*y);
51  double s = m_r/R;
52  // Scale back to the surface
53  p.x *= s; p.y *= s;
54  // Compute unit normal
55  double Nx = p.x/m_r, Ny = p.y/m_r;
56  // compute v.N
57  double v_dot_N = p.vx*Nx + p.vy*Ny;
58  // compute n.N
59  double n_dot_N = p.nx*Nx + p.ny*Ny;
60  // Project velocity onto tangent plane
61  p.vx -= v_dot_N*Nx; p.vy -= v_dot_N*Ny;
62  // Project director onto tangent plane
63  p.nx -= n_dot_N*Nx; p.ny -= n_dot_N*Ny;
64  // normalize director
65  double inv_len = 1.0/sqrt(p.nx*p.nx + p.ny*p.ny + p.nz*p.nz);
66  p.nx *= inv_len; p.ny *= inv_len; p.nz *= inv_len;
67  if (periodic)
68  {
69  if (p.z > box->zhi) p.z -= box->Lz;
70  else if (p.z < box->zlo) p.z += box->Lz;
71  }
72  }
73 }
