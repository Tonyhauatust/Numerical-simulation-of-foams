48  if (m_freq > 0 && t % m_freq == 0 && m_div_rate > 0.0) // Attempt division only at certain time steps
49  {
50  cout << "Handling divisions for group " << m_group_name << endl;
51  if (!m_system->group_ok(m_group_name))
52  {
53  cout << "Before divide P: Group info mismatch for group : " << m_group_name << endl;
54  throw runtime_error("Group mismatch.");
55  }
56  int new_type; // type of new particle
57  double new_r; // radius of newly formed particle
58  int N = m_system->get_group(m_group_name)->get_size();
59  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
60  BoxPtr box = m_system->get_box();
61  double prob_div = m_div_rate*m_freq*m_system->get_integrator_step(); // actual probability of dividing now: rate * (attempt_freq * dt)
62  if (prob_div > 1.0)
63  {
64  m_msg->msg(Messenger::ERROR,"Division rate "+lexical_cast<string>(prob_div)+" is too large for current time step and attempt rate.");
65  m_msg->msg(Messenger::INFO,"We have: division "+lexical_cast<string>(m_div_rate)+" frequency "+lexical_cast<string>(m_freq)+" time step "+lexical_cast<string>(m_system->get_step())+".");
66  throw runtime_error("Too high division.");
67  }
68  for (int i = 0; i < N; i++)
69  {
70  int pi = particles[i];
71  Particle& p = m_system->get_particle(pi);
72  if (m_rng->drnd() < prob_div*(1.0-p.coordination/m_rho_max))
73  {
74  cout << " dividing particle of type " << p.get_type() << endl;
75  Particle p_new(m_system->size(), p.get_type(), p.get_radius());
76  p_new.x = p.x + m_alpha*m_split_distance*p.get_radius()*p.nx;
77  p_new.y = p.y + m_alpha*m_split_distance*p.get_radius()*p.ny;
78  p_new.z = p.z + m_alpha*m_split_distance*p.get_radius()*p.nz;
79  p_new.set_parent(p.get_flag());
80  m_system->apply_periodic(p_new.x,p_new.y,p_new.z);
81 
82  p.x -= (1.0-m_alpha)*m_split_distance*p.get_radius()*p.nx;
83  p.y -= (1.0-m_alpha)*m_split_distance*p.get_radius()*p.ny;
84  p.z -= (1.0-m_alpha)*m_split_distance*p.get_radius()*p.nz;
85  m_system->apply_periodic(p.x,p.y,p.z);
86 
87  p_new.nx = p.nx; p_new.ny = p.ny; p_new.nz = p.nz;
88  p_new.vx = p.vx; p_new.vy = p.vy; p_new.vz = p.vz;
89  p_new.Nx = p.Nx; p_new.Ny = p.Ny; p_new.Nz = p.Nz;
90  p.age = 0.0; // age of parent is 0
91  p_new.age = 0.0; // age of child is 0
92  for(list<string>::iterator it_g = p.groups.begin(); it_g != p.groups.end(); it_g++)
93  p_new.add_group(*it_g);
94  if (p.in_tissue) p_new.in_tissue = true;
95  p_new.set_radius(p.get_radius());
96  p_new.set_length(p.get_length());
97  p_new.set_default_area(p.get_A0());
98  p_new.A0 = p.A0;
99  if (m_rng->drnd() < m_type_change_prob_1) // Attempt to change type and group for first child
100  {
101  if (m_new_type == 0)
102  new_type = p.get_type();
103  else
104  new_type = m_new_type;
105  p.set_type(new_type);
106  m_system->change_group(p.get_id(),m_old_group,m_new_group);
107  }
108  // For the polydispersity function: Change radius of second child.
109  if (m_new_radius == 0.0)
110  new_r = p_new.get_radius();
111  else
112  {
113  if (m_poly == 0.0)
114  new_r = m_new_radius;
115  else
116  new_r = m_new_radius*(1.0 + m_poly*(m_rng->drnd() - 0.5));
117  }
118  p_new.set_radius(new_r);
119  if (m_rng->drnd() < m_type_change_prob_2) // Attempt to change type and group for second child
120  {
121  if (m_new_type == 0)
122  new_type = p_new.get_type();
123  else
124  new_type = m_new_type;
125  p_new.set_type(new_type);
126  m_system->add_particle(p_new);
127  m_system->change_group(p_new.get_id(),m_old_group,m_new_group);
128  }
129  else
130  m_system->add_particle(p_new);
131  cout << "old particle: " << p << endl;
132  cout << "new particle: " << p << endl;
133  cout << "groups of new particle: " << endl;
134  for (list<string>::const_iterator it = p_new.groups.begin(); it != p_new.groups.end(); it++)
135  cout << format(" %s ") % (*it);
136  cout << endl << "groups of old particle: " << endl;
137  for (list<string>::const_iterator it = p.groups.begin(); it != p.groups.end(); it++)
138  cout << format(" %s ") % (*it);
139  //p.print_groups();
140  //p_new.print_groups();
141  }
142  }
143  if (!m_system->group_ok(m_group_name))
144  {
145  cout << "After Divide P: Group info mismatch for group : " << m_group_name << endl;
146  throw runtime_error("Group mismatch.");
147  }

160 void PopulationDensity::remove(int t)
161 {
162  if (m_freq > 0 && t % m_freq == 0 && m_death_rate > 0.0) // Attempt removal only at certain time steps
163  {
164  cout << "Handling deaths for group " << m_group_name << endl;
165  if (!m_system->group_ok(m_group_name))
166  {
167  cerr << "Before Remove P: Group info mismatch for group : " << m_group_name << endl;
168  throw runtime_error("Group mismatch.");
169  }
170  int N = m_system->get_group(m_group_name)->get_size();
171  vector<int> particles = m_system->get_group(m_group_name)->get_particles();
172  vector<int> to_remove;
173  double prob_death = m_death_rate*m_freq*m_system->get_integrator_step(); // actual probability of dividing now: rate * (attempt_freq * dt)
174  if (prob_death>1.0)
175  {
176  cerr << "Error: death rate " << prob_death << " is too large for current time step and attempt rate!" << endl;
177  throw runtime_error("Too high death.");
178  }
179  for (int i = 0; i < N; i++)
180  {
181  int pi = particles[i];
182  Particle& p = m_system->get_particle(pi);
183  if (m_rng->drnd() < prob_death) {
184  cout << "particle of type " << p.get_type() << " died " << endl;
185  to_remove.push_back(p.get_id());
186  }
187  }
188  int offset = 0;
189  for (vector<int>::iterator it = to_remove.begin(); it != to_remove.end(); it++)
190  {
191  m_system->remove_particle((*it)-offset);
192  offset++;
193  }
194  if (m_system->size() == 0)
195  {
196  m_msg->msg(Messenger::ERROR,"Density population control. No particles left in the system. Please reduce that death rate.");
197  throw runtime_error("No particles left in the system.");
198  }
199  if (!m_system->group_ok(m_group_name))
200  {
201  cout << "After Remove P: Group info mismatch for group : " << m_group_name << endl;
202  throw runtime_error("Group mismatch.");
203  }
204  m_system->set_force_nlist_rebuild(true);
205  }
